    ```php
    <?php
    declare(strict_types=1);
    namespace App\Http\Requests\FinnGen;

    use Illuminate\Foundation\Http\FormRequest;
    use Illuminate\Validation\Rule;

    final class TopVariantsQueryRequest extends FormRequest
    {
        public const ALLOWED_SORTS = ['chrom', 'pos', 'af', 'beta', 'se', 'p_value', 'snp_id'];

        public function authorize(): bool
        {
            return $this->user()?->can('finngen.workbench.use') ?? false;
        }

        /** @return array<string, array<int, mixed>> */
        public function rules(): array
        {
            return [
                'limit' => ['nullable', 'integer', 'min:1', 'max:200'],
                'sort' => ['nullable', 'string', Rule::in(self::ALLOWED_SORTS)],
                'dir' => ['nullable', 'string', Rule::in(['asc', 'desc', 'ASC', 'DESC'])],
            ];
        }
    }
    ```

    ```php
    <?php
    declare(strict_types=1);
    namespace App\Http\Controllers\Api\V1\FinnGen;

    use App\Http\Controllers\Controller;
    use App\Http\Requests\FinnGen\TopVariantsQueryRequest;
    use App\Models\App\FinnGen\Run;
    use App\Services\FinnGen\ManhattanAggregationService;
    use Illuminate\Http\JsonResponse;
    use Illuminate\Support\Facades\Cache;

    final class GwasTopVariantsController extends Controller
    {
        public function __construct(private readonly ManhattanAggregationService $agg) {}

        public function index(TopVariantsQueryRequest $request, string $runId): JsonResponse
        {
            /** @var Run|null $run */
            $run = Run::query()->find($runId);
            if ($run === null) abort(404);
            $user = $request->user();
            if ($run->user_id !== $user->id && ! $user->hasRole(['admin', 'super-admin'])) abort(403);

            if ($run->status === Run::STATUS_QUEUED || $run->status === Run::STATUS_RUNNING) {
                return response()->json(['status' => $run->status], 202)->header('Retry-After', '30');
            }
            if ($run->status === Run::STATUS_FAILED) abort(410, 'Run failed');
            if ($run->status !== Run::STATUS_SUCCEEDED) abort(409);

            $schema = $this->agg->resolveSchemaForRun($run);
            if ($schema === null) abort(404);

            $sort = (string) ($request->validated('sort') ?? 'p_value');
            $dir = strtoupper((string) ($request->validated('dir') ?? 'ASC'));
            $limit = (int) ($request->validated('limit') ?? 50);

            // Q6-RESOLVED: pull cohort_definition_id from Run.params if present (BTREE index speedup per Pitfall 6)
            /** @var array<string,mixed> $params */
            $params = $run->params ?? [];
            $cohortId = isset($params['cohort_definition_id']) ? (int) $params['cohort_definition_id'] : null;

            $key = sprintf('finngen:manhattan:%s:top-variants:%s:%s:%d', $runId, $sort, strtolower($dir), $limit);
            $payload = Cache::remember($key, now()->addMinutes(15), fn () =>
                $this->agg->topVariants($schema, $runId, $sort, $dir, $limit, $cohortId)
            );
            return response()->json($payload);
        }
    }
    ```
