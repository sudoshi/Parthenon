    ```ts
    import { test, expect } from "@playwright/test";

    // SC-1 perf: warm-cache render must complete within 3 seconds.
    // Q4 acknowledgment: measured against Phase 14-05 10k-row PANCREAS smoke run (no 10M corpus exists on DEV).
    //   Expected cold-cache first-hit: ~5-10s; warm-cache: ~500ms. Target: <3s warm.

    const RUN_ID = process.env.PHASE_16_RUN_ID ?? "01JXXXXXXXXXXXXXXXXXXXXXXX";
    const ENDPOINT_NAME = process.env.PHASE_16_ENDPOINT ?? "E4_DM2";

    test.describe("Phase 16 Manhattan perf", () => {
        test.beforeEach(async ({ page }) => {
            // Auth fixture: login as admin@acumenus.net via existing Playwright harness
            await page.goto("/login");
            await page.fill('input[name="email"]', process.env.ADMIN_EMAIL ?? "admin@acumenus.net");
            await page.fill('input[name="password"]', process.env.ADMIN_PASSWORD ?? "");
            await page.click('button[type="submit"]');
            await page.waitForURL(/\/workbench|\/$/);
        });

        test("warm-cache Manhattan renders within 3 seconds", async ({ page }) => {
            // First visit: warms Redis cache (ignore latency)
            await page.goto(`/workbench/finngen-endpoints/${ENDPOINT_NAME}/gwas/${RUN_ID}`);
            await page.waitForSelector("canvas[role='img'][aria-label*='Manhattan']", { timeout: 30_000 });

            // Second visit: warm cache — this is the measurement
            const t0 = performance.now();
            await page.goto(`/workbench/finngen-endpoints/${ENDPOINT_NAME}/gwas/${RUN_ID}`);
            await page.waitForSelector("canvas[role='img'][aria-label*='Manhattan']", { timeout: 5_000 });
            // Wait for first 100 dots drawn — proxied by thinning banner showing variant_count_after
            await page.waitForSelector("[data-thinning-banner], text=/Displaying .+ of .+ variants/", { timeout: 3_000 });
            const elapsed = performance.now() - t0;

            console.log(`SC-1 warm-cache render: ${elapsed.toFixed(0)}ms`);
            expect(elapsed).toBeLessThan(3_000);
        });

        test("top-variants table renders with 50 rows", async ({ page }) => {
            await page.goto(`/workbench/finngen-endpoints/${ENDPOINT_NAME}/gwas/${RUN_ID}`);
            await page.waitForSelector("table", { timeout: 10_000 });
            const rowCount = await page.locator("table tbody tr").count();
            expect(rowCount).toBeGreaterThan(0);
            expect(rowCount).toBeLessThanOrEqual(50);
        });
    });
    ```
