    <?php
    declare(strict_types=1);
    namespace App\Services\FinnGen;

    use App\Models\App\FinnGen\Run;
    use App\Models\App\Source;
    use Illuminate\Support\Facades\DB;

    final class ManhattanAggregationService
    {
        private const CASE_COHORT_PARAM_KEY = 'cohort_definition_id';  // Q6 RESOLVED via grep
        private const SCHEMA_WHITELIST_RE = '/^[a-z][a-z0-9_]*$/';

        public function resolveSchemaForRun(Run $run): ?string
        {
            $sourceKey = (string) $run->source_key;
            if (! Source::query()->where('source_key', $sourceKey)->exists()) return null;
            $schema = strtolower($sourceKey) . '_gwas_results';
            if (preg_match(self::SCHEMA_WHITELIST_RE, $schema) !== 1) return null;
            $hasSchema = DB::selectOne(
                'SELECT 1 AS ok FROM information_schema.schemata WHERE schema_name = ? LIMIT 1',
                [$schema]
            );
            return $hasSchema === null ? null : $schema;
        }

        /** @return array{variants: list<array{chrom:string,pos:int,neg_log_p:float,snp_id:?string}>, genome: array{chrom_offsets: array<string,int>}, thinning: array{bins:int,threshold:float,variant_count_before:int,variant_count_after:int}} */
        public function thin(string $schema, string $runId, int $binCount, float $threshold): array
        {
            // Defense in depth: re-validate schema against whitelist even though caller did.
            if (preg_match(self::SCHEMA_WHITELIST_RE, $schema) !== 1) {
                throw new \InvalidArgumentException("Invalid schema: $schema");
            }

            $countBefore = (int) DB::scalar(
                "SELECT COUNT(*) FROM {$schema}.summary_stats WHERE gwas_run_id = ?",
                [$runId]
            );

            $sql = <<<SQL
WITH per_chrom_bounds AS (
    SELECT chrom, MIN(pos)::bigint AS chrom_start, MAX(pos)::bigint AS chrom_end
      FROM {$schema}.summary_stats
     WHERE gwas_run_id = ?
  GROUP BY chrom
),
binned AS (
    SELECT ss.chrom, ss.pos, ss.p_value, ss.snp_id,
           ROW_NUMBER() OVER (
               PARTITION BY ss.chrom,
                            width_bucket(ss.pos, b.chrom_start, b.chrom_end + 1, ?)
               ORDER BY ss.p_value ASC NULLS LAST, ss.pos ASC
           ) AS rnk
      FROM {$schema}.summary_stats ss
      JOIN per_chrom_bounds b ON b.chrom = ss.chrom
     WHERE ss.gwas_run_id = ?
),
thin_reps AS (SELECT chrom, pos, p_value, snp_id FROM binned WHERE rnk = 1),
gws_bypass AS (
    SELECT chrom, pos, p_value, snp_id
      FROM {$schema}.summary_stats
     WHERE gwas_run_id = ? AND p_value < ? AND p_value IS NOT NULL
)
SELECT chrom, pos, p_value, snp_id FROM thin_reps
UNION
SELECT chrom, pos, p_value, snp_id FROM gws_bypass
ORDER BY chrom, pos
SQL;

            $rows = DB::select($sql, [$runId, $binCount, $runId, $runId, $threshold]);

            $variants = array_map(fn($r) => [
                'chrom' => (string) $r->chrom,
                'pos' => (int) $r->pos,
                'neg_log_p' => $r->p_value !== null ? -log10((float) $r->p_value) : 0.0,
                'snp_id' => $r->snp_id !== null ? (string) $r->snp_id : null,
            ], $rows);

            // chrom_offsets: cumulative chrom-end offsets for Manhattan x-axis genome coords
            $offsets = [];
            $cum = 0;
            foreach (DB::select(
                "SELECT chrom, MAX(pos)::bigint AS chrom_end FROM {$schema}.summary_stats WHERE gwas_run_id = ? GROUP BY chrom ORDER BY chrom",
                [$runId]
            ) as $o) {
                $offsets[(string) $o->chrom] = $cum;
                $cum += (int) $o->chrom_end;
            }

            return [
                'variants' => $variants,
                'genome' => ['chrom_offsets' => $offsets],
                'thinning' => [
                    'bins' => $binCount,
                    'threshold' => $threshold,
                    'variant_count_before' => $countBefore,
                    'variant_count_after' => count($variants),
                ],
            ];
        }

        /** @return array{variants: list<array<string,mixed>>, chrom: string, start: int, end: int} */
        public function region(string $schema, string $runId, string $chrom, int $start, int $end): array
        {
            if (preg_match(self::SCHEMA_WHITELIST_RE, $schema) !== 1) {
                throw new \InvalidArgumentException("Invalid schema: $schema");
            }
            $rows = DB::select(
                "SELECT chrom, pos, ref, alt, af, beta, se, p_value, snp_id
                 FROM {$schema}.summary_stats
                 WHERE gwas_run_id = ? AND chrom = ? AND pos BETWEEN ? AND ?
                 ORDER BY pos ASC",
                [$runId, $chrom, $start, $end]
            );
            return [
                'variants' => array_map(fn($r) => (array) $r, $rows),
                'chrom' => $chrom, 'start' => $start, 'end' => $end,
            ];
        }

        /** @return array{rows: list<array<string,mixed>>, total: int} */
        public function topVariants(string $schema, string $runId, string $sort, string $dir, int $limit, ?int $cohortDefinitionId = null): array
        {
            if (preg_match(self::SCHEMA_WHITELIST_RE, $schema) !== 1) {
                throw new \InvalidArgumentException("Invalid schema: $schema");
            }
            $allowedSorts = ['chrom','pos','af','beta','se','p_value','snp_id'];
            if (! in_array($sort, $allowedSorts, true)) {
                throw new \InvalidArgumentException("Invalid sort column: $sort");
            }
            $dir = strtoupper($dir) === 'DESC' ? 'DESC' : 'ASC';
            $cohortClause = $cohortDefinitionId !== null ? 'AND cohort_definition_id = ?' : '';
            $params = $cohortDefinitionId !== null ? [$runId, $cohortDefinitionId, $limit] : [$runId, $limit];

            $rows = DB::select(
                "SELECT chrom, pos, ref, alt, af, beta, se, p_value, snp_id, gwas_run_id
                 FROM {$schema}.summary_stats
                 WHERE gwas_run_id = ? {$cohortClause}
                 ORDER BY {$sort} {$dir} NULLS LAST, pos ASC
                 LIMIT ?",
                $params
            );
            return ['rows' => array_map(fn($r) => (array) $r, $rows), 'total' => count($rows)];
        }
    }

    <?php
    declare(strict_types=1);
    namespace App\Services\FinnGen;

    final class GencodeService
    {
        /** @var array<int, array{gene_name:string,chrom:string,start:int,end:int,strand:string,gene_type:string}>|null */
        private static ?array $genes = null;

        public const DEFAULT_TYPES = ['protein_coding', 'lincRNA', 'lncRNA', 'miRNA'];

        /** @return list<array{gene_name:string,chrom:string,start:int,end:int,strand:string,gene_type:string}> */
        public function findGenesInRange(string $chrom, int $start, int $end, bool $includePseudogenes = false): array
        {
            self::$genes ??= $this->load();
            $out = [];
            foreach (self::$genes as $g) {
                if ($g['chrom'] !== $chrom) continue;
                if ($g['start'] > $end || $g['end'] < $start) continue;
                if (! $includePseudogenes && ! in_array($g['gene_type'], self::DEFAULT_TYPES, true)) continue;
                $out[] = $g;
            }
            return $out;
        }

        /** @return array<int, array{gene_name:string,chrom:string,start:int,end:int,strand:string,gene_type:string}> */
        private function load(): array
        {
            $path = storage_path('app/private/gencode/genes-v46.tsv');
            if (! is_file($path)) return [];
            $rows = [];
            $fh = fopen($path, 'rb');
            try {
                while (($line = fgets($fh)) !== false) {
                    $c = explode("\t", rtrim($line, "\r\n"));
                    if (count($c) < 6) continue;
                    $rows[] = [
                        'gene_name' => $c[0], 'chrom' => $c[1],
                        'start' => (int) $c[2], 'end' => (int) $c[3],
                        'strand' => $c[4], 'gene_type' => $c[5],
                    ];
                }
            } finally { fclose($fh); }
            return $rows;
        }
    }

    <?php
    declare(strict_types=1);
    namespace App\Console\Commands\FinnGen;

    use Illuminate\Console\Command;
    use Illuminate\Support\Facades\Http;

    final class LoadGencodeGtfCommand extends Command
    {
        protected $signature = 'parthenon:load-gencode-gtf
            {--force : Re-download/re-parse even if TSV exists}
            {--file= : Local GFF3.gz path under storage/ (testing/dev override; bypasses SSRF)}';
        protected $description = 'Ingest GENCODE v46 basic annotation GFF3 → storage/app/private/gencode/genes-v46.tsv';

        private const URL = 'https://ftp.ebi.ac.uk/pub/databases/gencode/Gencode_human/release_46/gencode.v46.basic.annotation.gff3.gz';
        private const MAX_BYTES = 100 * 1024 * 1024;  // D-29 size guard
        private const OUTPUT_TSV = 'gencode/genes-v46.tsv';
        private const MAX_AGE_DAYS = 30;

        public function handle(): int
        {
            $outputPath = storage_path('app/private/' . self::OUTPUT_TSV);

            // Idempotency: skip if TSV exists recent & !--force
            if (! $this->option('force') && is_file($outputPath)) {
                $ageDays = (time() - filemtime($outputPath)) / 86400;
                if ($ageDays < self::MAX_AGE_DAYS) {
                    $this->info("TSV exists ({$outputPath}), age " . round($ageDays, 1) . "d < " . self::MAX_AGE_DAYS . "d. Use --force to overwrite.");
                    return self::SUCCESS;
                }
            }

            // SSRF-safe source resolution
            $gzPath = $this->resolveSourcePath();
            if (filesize($gzPath) > self::MAX_BYTES) {
                throw new \RuntimeException(sprintf('GFF3 exceeds %d-byte D-29 limit', self::MAX_BYTES));
            }

            @mkdir(dirname($outputPath), 0755, true);
            $tmp = $outputPath . '.tmp';
            $out = fopen($tmp, 'wb');
            $parsed = 0;
            $fh = @gzopen($gzPath, 'rb');
            if ($fh === false) throw new \RuntimeException("Cannot open gz: $gzPath");
            try {
                while (($line = gzgets($fh)) !== false) {
                    if ($line === '' || $line[0] === '#') continue;
                    $cols = explode("\t", rtrim($line, "\r\n"));
                    if (count($cols) < 9 || $cols[2] !== 'gene') continue;
                    $name = $this->parseAttr($cols[8], 'gene_name');
                    if ($name === null) continue;
                    $type = $this->parseAttr($cols[8], 'gene_type') ?? '';
                    $chrom = ltrim($cols[0], 'chr');
                    if ($chrom === 'M') $chrom = 'MT';
                    fwrite($out, implode("\t", [$name, $chrom, $cols[3], $cols[4], $cols[6], $type]) . "\n");
                    $parsed++;
                }
            } finally {
                gzclose($fh);
                fclose($out);
            }
            rename($tmp, $outputPath);
            $this->info("Wrote {$parsed} gene rows to {$outputPath}");
            return self::SUCCESS;
        }

        private function resolveSourcePath(): string
        {
            $localFile = $this->option('file');
            if ($localFile !== null) {
                $real = realpath((string) $localFile);
                if ($real === false) throw new \RuntimeException("--file not found: $localFile");
                $allowedRoot = realpath(storage_path()) ?: storage_path();
                if (! str_starts_with($real, $allowedRoot)) {
                    throw new \RuntimeException("--file must live under storage_path() (SSRF guard T-16-S6)");
                }
                return $real;
            }
            $tmp = tempnam(sys_get_temp_dir(), 'gencode_');
            $resp = Http::timeout(600)->withOptions(['sink' => $tmp, 'verify' => true])->get(self::URL);
            if (! $resp->successful()) {
                @unlink($tmp);
                throw new \RuntimeException("GENCODE fetch failed: HTTP {$resp->status()}");
            }
            return $tmp;
        }

        private function parseAttr(string $attrs, string $key): ?string
        {
            if (preg_match('/(?:^|;)\s*' . preg_quote($key, '/') . '=([^;]+)/', $attrs, $m) === 1) {
                return trim($m[1]);
            }
            return null;
        }
    }
