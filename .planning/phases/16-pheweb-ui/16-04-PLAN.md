    // useManhattanData.ts
    import { useQuery } from "@tanstack/react-query";
    import { fetchManhattan, type ManhattanPayload, type ManhattanInFlightResponse } from "../api/gwas-results";

    export function useManhattanData(runId: string, binCount = 100) {
        return useQuery({
            queryKey: ["finngen", "manhattan", runId, binCount],
            queryFn: () => fetchManhattan(runId, binCount),
            enabled: Boolean(runId),
            staleTime: 24 * 60 * 60 * 1000,  // 24h matches server TTL (D-20)
            retry: (failureCount, error: unknown) => {
                // Retry on network errors, NOT on 404/403/422
                const status = (error as { response?: { status?: number } }).response?.status;
                if (status !== undefined && [404, 403, 422, 410, 409].includes(status)) return false;
                return failureCount < 3;
            },
            refetchInterval: (query) => {
                // In-flight 202 → poll every 30s (matches Retry-After)
                const data = query.state.data;
                if (data && "status" in (data as object) && ((data as ManhattanInFlightResponse).status === "queued" || (data as ManhattanInFlightResponse).status === "running")) {
                    return 30_000;
                }
                return false;
            },
        });
    }

    export function isManhattanInFlight(data: ManhattanPayload | ManhattanInFlightResponse | undefined): data is ManhattanInFlightResponse {
        return data !== undefined && "status" in data;
    }
    ```

    import { useMemo } from "react";
    import { useManhattanData, isManhattanInFlight } from "../../hooks/useManhattanData";
    import type { ManhattanPayload, ManhattanVariant } from "../../api/gwas-results";
    import ManhattanPlot from "../../../investigation/components/genomic/ManhattanPlot";
    import { ErrorBoundary } from "../../../../components/ErrorBoundary";
    import { EmptyState } from "../../../../components/ui/EmptyState";

    export interface FinnGenManhattanPanelProps {
        runId: string;
        onPeakClick?: (chrom: string, pos: number) => void;
        width?: number;
        height?: number;
    }

    export function FinnGenManhattanPanel({ runId, onPeakClick, width = 1200, height = 400 }: FinnGenManhattanPanelProps): JSX.Element {
        const { data, isLoading, isError, error } = useManhattanData(runId, 100);

        if (isLoading) return <EmptyState title="Loading Manhattan plot…" />;
        if (isError) {
            const status = (error as { response?: { status?: number } }).response?.status;
            if (status === 410) return <EmptyState title="Run failed" description="The GWAS run did not complete. Check the run log." />;
            if (status === 404) return <EmptyState title="Run not found" />;
            return <EmptyState title="Unable to load Manhattan plot" description={(error as Error).message} />;
        }
        if (data && isManhattanInFlight(data)) {
            return <EmptyState title="GWAS run is still processing" description={`Status: ${data.status}. Polling every 30s.`} />;
        }
        const payload = data as ManhattanPayload;
        const points = useMemo(() => payload.variants.map((v: ManhattanVariant) => ({
            chr: v.chrom, pos: v.pos, p: Math.pow(10, -v.neg_log_p), negLogP: v.neg_log_p, snpId: v.snp_id ?? undefined,
        })), [payload.variants]);

        return (
            <ErrorBoundary fallback={<EmptyState title="Chart crashed" description="Reload to retry" />}>
                <div className="rounded-lg border border-border bg-surface p-4">
                    <div className="mb-2 flex items-center justify-between text-xs text-text-muted">
                        <span>
                            Displaying {payload.thinning.variant_count_after.toLocaleString()} of {payload.thinning.variant_count_before.toLocaleString()} variants
                            (threshold p&lt;{payload.thinning.threshold.toExponential(1)})
                        </span>
                    </div>
                    <ManhattanPlot
                        data={points}
                        preThinned
                        width={width}
                        height={height}
                        onPointClick={(p) => onPeakClick?.(p.chr, p.pos)}
                    />
                </div>
            </ErrorBoundary>
        );
    }
    ```
