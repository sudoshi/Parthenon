```typescript
export async function generateEndpoint(
  name: string,
  payload: GenerateEndpointPayload,
): Promise<GenerateEndpointResponse> {
  try {
    const r = await apiClient.post<GenerateEndpointResponse>(
      `/finngen/endpoints/${encodeURIComponent(name)}/generate`,
      payload,
    );
    return r.data;
  } catch (err: unknown) {
    // Re-throw preserving the 422 body for tooltip copy.
    if (typeof err === "object" && err !== null && "response" in err) {
      const r = (err as { response?: { status?: number; data?: unknown } }).response;
      if (r?.status === 422 && r?.data) {
        throw r.data;
      }
    }
    throw err;
  }
}
```

  profile,
  className,

    return (
      <span
        data-testid="coverage-profile-badge-finland-only"

        title="This endpoint depends on Finnish-specific source vocabularies (ICD-8, NOMESCO, KELA_REIMB, ICDO3-FI)."

      </span>
