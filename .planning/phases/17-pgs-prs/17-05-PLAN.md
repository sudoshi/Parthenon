```typescript
// Phase 17 GENOMICS-08 — frontend API shims for the cohort PRS endpoints.
import { api } from '@/lib/api';

export type PrsHistogramBin = {
  bin: number;
  bin_lo: number;
  bin_hi: number;
  n: number;
};

export type PrsSummary = {
  mean: number | null;
  stddev: number | null;
  min: number | null;
  max: number | null;
  median: number | null;
  iqr_q1: number | null;
  iqr_q3: number | null;
};

export type PrsQuintiles = {
  q20: number | null;
  q40: number | null;
  q60: number | null;
  q80: number | null;
};

export type PrsScoreResult = {
  score_id: string;
  pgs_name: string | null;
  trait_reported: string | null;
  scored_at: string;
  subject_count: number;
  summary: PrsSummary;
  quintiles: PrsQuintiles;
  histogram: PrsHistogramBin[];
};

export type CohortPrsResponse = { scores: PrsScoreResult[] };

export type PgsCatalogScore = {
  score_id: string;
  pgs_name: string | null;
  trait_reported: string | null;
  variants_number: number | null;
};

export type PgsCatalogScoresResponse = { scores: PgsCatalogScore[] };

export type ComputePrsRequest = {
  source_key: string;
  score_id: string;
  cohort_definition_id?: number;
  overwrite_existing?: boolean;
};

export type ComputePrsResponse = {
  data: {
    run: { id: string; status: string };
    analysis_type: 'finngen.prs.compute';
    cohort_definition_id: number;
    score_id: string;
    source_key: string;
    finngen_endpoint_generation_id: number | null;
  };
};

export async function fetchCohortPrsScores(cohortId: number, bins = 50): Promise<CohortPrsResponse> {
  const { data } = await api.get<CohortPrsResponse>(`/cohort-definitions/${cohortId}/prs`, {
    params: { bins },
  });
  return data;
}

export async function fetchPgsCatalogScores(): Promise<PgsCatalogScoresResponse> {
  const { data } = await api.get<PgsCatalogScoresResponse>('/pgs-catalog/scores');
  return data;
}

export async function dispatchComputePrs(endpointName: string, req: ComputePrsRequest): Promise<ComputePrsResponse> {
  const { data } = await api.post<ComputePrsResponse>(`/finngen/endpoints/${endpointName}/prs`, req);
  return data;
}

export function buildPrsDownloadUrl(cohortId: number, scoreId: string): string {
  // Absolute URL relative to the Axios baseURL; caller typically uses this in a plain <a href>
  // so the browser handles the streaming download + Content-Disposition attachment naturally.
  return `${api.defaults.baseURL ?? ''}/cohort-definitions/${cohortId}/prs/${scoreId}/download`;
}
```

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import {
  fetchCohortPrsScores,
  fetchPgsCatalogScores,
  dispatchComputePrs,
  type ComputePrsRequest,
} from '../api/prs';

export function useCohortPrsScores(cohortId: number, bins = 50) {
  return useQuery({
    queryKey: ['cohort', cohortId, 'prs', bins],
    queryFn: () => fetchCohortPrsScores(cohortId, bins),
    enabled: Number.isFinite(cohortId) && cohortId > 0,
  });
}

export function usePgsCatalogScores() {
  return useQuery({
    queryKey: ['pgs-catalog', 'scores'],
    queryFn: fetchPgsCatalogScores,
    staleTime: 5 * 60 * 1000,
  });
}

export function useComputePrsMutation(endpointName: string, cohortId: number) {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: (req: ComputePrsRequest) => dispatchComputePrs(endpointName, req),
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: ['cohort', cohortId, 'prs'] });
    },
  });
}
```

```tsx
// Phase 17 GENOMICS-08 — cohort drawer PRS distribution panel.
// Renders Recharts BarChart + 5 quintile ReferenceArea bands + summary stats + CSV download.
// Empty state renders 'Compute PRS' CTA handed off to parent via onCompute.
import { useState, useMemo } from 'react';
import {
  BarChart, Bar, XAxis, YAxis, CartesianGrid, Tooltip,
  ReferenceArea, ResponsiveContainer,
} from 'recharts';
import { useCohortPrsScores } from '../hooks/usePrsScores';
import { buildPrsDownloadUrl, type PrsScoreResult } from '../api/prs';

type Props = {
  cohortId: number;
  bins?: number;
  onCompute?: () => void;
};

function fmt(v: number | null | undefined, digits = 3): string {
  if (v === null || v === undefined || !Number.isFinite(v)) return '—';
  return v.toPrecision(digits);
}

function buildBands(score: PrsScoreResult) {
  const min = score.summary.min ?? 0;
  const max = score.summary.max ?? 1;
  const q = score.quintiles;
  const band = (x1: number | null, x2: number | null, opacity: number) =>
    x1 !== null && x2 !== null ? { x1, x2, opacity } : null;
  return [
    band(min, q.q20, 0.10),
    band(q.q20, q.q40, 0.20),
    band(q.q40, q.q60, 0.30),  // median band
    band(q.q60, q.q80, 0.20),
    band(q.q80, max, 0.10),
  ].filter((b): b is { x1: number; x2: number; opacity: number } => b !== null);
}

function buildHistogramData(score: PrsScoreResult) {
  // Bar chart needs bin_mid as the x-value since bars are centered.
  return score.histogram.map(h => ({
    bin_mid: (h.bin_lo + h.bin_hi) / 2,
    count: h.n,
    bin_lo: h.bin_lo,
    bin_hi: h.bin_hi,
  }));
}

export function PrsDistributionPanel({ cohortId, bins = 50, onCompute }: Props) {
  const { data, isLoading, error } = useCohortPrsScores(cohortId, bins);
  const scores = data?.scores ?? [];
  const [selectedScoreId, setSelectedScoreId] = useState<string>('');

  const selected: PrsScoreResult | undefined = useMemo(
    () => scores.find(s => s.score_id === selectedScoreId) ?? scores[0],
    [scores, selectedScoreId],
  );

  if (isLoading) {
    return (
      <div role="status" className="p-6 text-sm text-[color:var(--text-secondary)]">
        Loading PRS scores...
      </div>
    );
  }
  if (error) {
    return (
      <div role="alert" className="p-6 text-sm text-[color:var(--color-crimson)]">
        Error loading PRS scores: {(error as Error).message}
      </div>
    );
  }
  if (scores.length === 0) {
    return (
      <div className="p-6 rounded border border-[color:var(--border-default)] text-center">
        <p className="text-sm mb-3 text-[color:var(--text-secondary)]">
          No polygenic risk scores computed for this cohort yet.
        </p>
        <button
          type="button"
          onClick={onCompute}
          className="px-4 py-2 rounded bg-[color:var(--color-crimson)] text-white text-sm hover:opacity-90"
        >
          Compute PRS
        </button>
      </div>
    );
  }

  const bands = selected ? buildBands(selected) : [];
  const histogramData = selected ? buildHistogramData(selected) : [];

  return (
    <section aria-label="PRS distribution" className="space-y-4">
      <header className="flex items-center justify-between gap-3">
        <label className="text-sm flex items-center gap-2">
          <span className="text-[color:var(--text-secondary)]">Score:</span>
          <select
            aria-label="Select PRS score"
            value={selected?.score_id ?? ''}
            onChange={e => setSelectedScoreId(e.target.value)}
            className="px-2 py-1 rounded bg-[color:var(--bg-surface)] border border-[color:var(--border-default)]"
          >
            {scores.map(s => (
              <option key={s.score_id} value={s.score_id}>
                {s.trait_reported ?? '(no trait)'} — {s.score_id}
              </option>
            ))}
          </select>
        </label>
        {selected && (
          <a
            href={buildPrsDownloadUrl(cohortId, selected.score_id)}
            target="_blank"
            rel="noreferrer"
            className="text-sm underline text-[color:var(--color-teal)]"
          >
            Download CSV
          </a>
        )}
      </header>

      {selected && (
        <>
          <div className="h-[280px] w-full">
            <ResponsiveContainer width="100%" height={280}>
              <BarChart data={histogramData} margin={{ top: 16, right: 16, bottom: 32, left: 8 }}>
                <CartesianGrid strokeDasharray="3 3" stroke="var(--border-default)" />
                <XAxis
                  dataKey="bin_mid"
                  type="number"
                  domain={[selected.summary.min ?? 0, selected.summary.max ?? 1]}
                  scale="linear"
                  stroke="var(--text-secondary)"
                  fontSize={11}
                />
                <YAxis stroke="var(--text-secondary)" fontSize={11} />
                {bands.map((b, i) => (
                  <ReferenceArea
                    key={i}
                    x1={b.x1}
                    x2={b.x2}
                    fill="var(--color-crimson)"
                    fillOpacity={b.opacity}
                  />
                ))}
                <Bar dataKey="count" fill="var(--color-teal)" />
                <Tooltip formatter={((v: number) => [`${v}`, 'subjects']) as never} />
              </BarChart>
            </ResponsiveContainer>
          </div>

          <dl className="grid grid-cols-2 sm:grid-cols-4 gap-3 text-sm">
            <div><dt className="text-[color:var(--text-secondary)]">Subjects</dt><dd>{selected.subject_count}</dd></div>
            <div><dt className="text-[color:var(--text-secondary)]">Mean</dt><dd>{fmt(selected.summary.mean)}</dd></div>
            <div><dt className="text-[color:var(--text-secondary)]">Median</dt><dd>{fmt(selected.summary.median)}</dd></div>
            <div><dt className="text-[color:var(--text-secondary)]">SD</dt><dd>{fmt(selected.summary.stddev)}</dd></div>
            <div><dt className="text-[color:var(--text-secondary)]">Min</dt><dd>{fmt(selected.summary.min)}</dd></div>
            <div><dt className="text-[color:var(--text-secondary)]">Max</dt><dd>{fmt(selected.summary.max)}</dd></div>
            <div><dt className="text-[color:var(--text-secondary)]">IQR Q1</dt><dd>{fmt(selected.summary.iqr_q1)}</dd></div>
            <div><dt className="text-[color:var(--text-secondary)]">IQR Q3</dt><dd>{fmt(selected.summary.iqr_q3)}</dd></div>
          </dl>
        </>
      )}
    </section>
  );
}
```

```tsx
import { useState } from 'react';
import { usePgsCatalogScores, useComputePrsMutation } from '../hooks/usePrsScores';

type Props = {
  open: boolean;
  onClose: () => void;
  cohortId: number;
  endpointName: string | null;
  sourceKey: string | null;
};

export function ComputePrsModal({ open, onClose, cohortId, endpointName, sourceKey }: Props) {
  const [scoreId, setScoreId] = useState<string>('');
  const [sourceKeyInput, setSourceKeyInput] = useState<string>(sourceKey ?? '');
  const [errMsg, setErrMsg] = useState<string | null>(null);

  const { data: catalog, isLoading } = usePgsCatalogScores();
  const mutation = useComputePrsMutation(endpointName ?? '', cohortId);

  if (!open) return null;

  const canSubmit =
    endpointName !== null && endpointName !== '' &&
    scoreId !== '' && sourceKeyInput !== '' && !mutation.isPending;

  async function onSubmit(e: React.FormEvent) {
    e.preventDefault();
    setErrMsg(null);
    if (!canSubmit || endpointName === null) return;
    try {
      await mutation.mutateAsync({
        source_key: sourceKeyInput,
        score_id: scoreId,
        cohort_definition_id: cohortId,
      });
      onClose();
    } catch (err) {
      setErrMsg((err as Error).message ?? 'Compute PRS failed');
    }
  }

  return (
    <div role="dialog" aria-modal="true" aria-labelledby="compute-prs-title"
         className="fixed inset-0 bg-black/60 flex items-center justify-center z-50">
      <form onSubmit={onSubmit}
            className="bg-[color:var(--bg-surface)] p-6 rounded w-[480px] max-w-[90vw] space-y-4">
        <h2 id="compute-prs-title" className="text-lg font-semibold">Compute PRS</h2>

        {endpointName === null && (
          <p role="alert" className="text-sm text-[color:var(--color-crimson)]">
            PRS compute is available only for FinnGen endpoint cohorts in v1. Materialize this
            cohort against a FinnGen endpoint first.
          </p>
        )}

        <label className="block text-sm">
          <span className="text-[color:var(--text-secondary)]">PGS Catalog score</span>
          <select
            aria-label="PGS Catalog score"
            value={scoreId}
            onChange={e => setScoreId(e.target.value)}
            className="mt-1 block w-full px-2 py-1 rounded bg-[color:var(--bg-default)] border border-[color:var(--border-default)]"
            disabled={isLoading || endpointName === null}
          >
            <option value="">{isLoading ? 'Loading...' : '— select —'}</option>
            {(catalog?.scores ?? []).map(s => (
              <option key={s.score_id} value={s.score_id}>
                {s.trait_reported ?? '(no trait)'} — {s.score_id}
                {s.variants_number ? ` (${s.variants_number} variants)` : ''}
              </option>
            ))}
          </select>
        </label>

        <label className="block text-sm">
          <span className="text-[color:var(--text-secondary)]">Source key (e.g. PANCREAS)</span>
          <input
            type="text"
            aria-label="Source key"
            value={sourceKeyInput}
            onChange={e => setSourceKeyInput(e.target.value.toUpperCase())}
            className="mt-1 block w-full px-2 py-1 rounded bg-[color:var(--bg-default)] border border-[color:var(--border-default)]"
            pattern="^[A-Z][A-Z0-9_]*$"
            disabled={endpointName === null}
          />
        </label>

        {errMsg && <p role="alert" className="text-sm text-[color:var(--color-crimson)]">{errMsg}</p>}

        <div className="flex justify-end gap-2 pt-2">
          <button type="button" onClick={onClose}
                  className="px-3 py-1 text-sm rounded border border-[color:var(--border-default)]">
            Cancel
          </button>
          <button type="submit" disabled={!canSubmit}
                  title={endpointName === null ? 'Available only for FinnGen endpoint cohorts in v1' : undefined}
                  className="px-3 py-1 text-sm rounded bg-[color:var(--color-crimson)] text-white disabled:opacity-50">
            {mutation.isPending ? 'Submitting...' : 'Compute PRS'}
          </button>
        </div>
      </form>
    </div>
  );
}
```
