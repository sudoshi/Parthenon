```sql
-- Histogram
WITH
bounds AS (
    SELECT MIN(raw_score) AS lo, MAX(raw_score) AS hi
      FROM {schema}.prs_subject_scores
     WHERE score_id = :score_id AND cohort_definition_id = :cohort_id
),
binned AS (
    SELECT width_bucket(raw_score, bounds.lo, bounds.hi, :bins) AS bin, raw_score
      FROM {schema}.prs_subject_scores, bounds
     WHERE score_id = :score_id AND cohort_definition_id = :cohort_id
)
SELECT LEAST(bin, :bins)::int AS bin,
       COUNT(*)::int AS n,
       MIN(raw_score)::float AS bin_lo,
       MAX(raw_score)::float AS bin_hi
  FROM binned
 GROUP BY LEAST(bin, :bins)
 ORDER BY bin;

-- Summary + quintiles
SELECT
    AVG(raw_score)::float AS mean,
    STDDEV(raw_score)::float AS stddev,
    MIN(raw_score)::float AS min,
    MAX(raw_score)::float AS max,
    percentile_cont(0.20) WITHIN GROUP (ORDER BY raw_score)::float AS q20,
    percentile_cont(0.40) WITHIN GROUP (ORDER BY raw_score)::float AS q40,
    percentile_cont(0.50) WITHIN GROUP (ORDER BY raw_score)::float AS median,
    percentile_cont(0.60) WITHIN GROUP (ORDER BY raw_score)::float AS q60,
    percentile_cont(0.80) WITHIN GROUP (ORDER BY raw_score)::float AS q80,
    percentile_cont(0.25) WITHIN GROUP (ORDER BY raw_score)::float AS iqr_q1,
    percentile_cont(0.75) WITHIN GROUP (ORDER BY raw_score)::float AS iqr_q3,
    COUNT(*)::int AS subject_count
  FROM {schema}.prs_subject_scores
 WHERE score_id = :score_id AND cohort_definition_id = :cohort_id;
```

**Step 1: `backend/app/Services/FinnGen/PrsAggregationService.php`** — service wraps the two SQL queries from RESEARCH §Pattern 3 verbatim:

```php
<?php

declare(strict_types=1);

namespace App\Services\FinnGen;

use Illuminate\Support\Facades\DB;

/**
 * Phase 17 GENOMICS-08 D-05 — aggregates per-cohort PRS distributions server-side
 * so the frontend never sees raw per-subject scores (T-17-S2 Information-Disclosure
 * mitigation). Uses PG width_bucket + percentile_cont per 17-RESEARCH §Pattern 3.
 */
final class PrsAggregationService
{
    private const SAFE_SCHEMA_REGEX = '/^[a-z][a-z0-9_]*$/';

    /**
     * @return array{
     *   subject_count: int,
     *   summary: array{mean:?float, stddev:?float, min:?float, max:?float, median:?float, iqr_q1:?float, iqr_q3:?float},
     *   quintiles: array{q20:?float, q40:?float, q60:?float, q80:?float},
     *   histogram: list<array{bin:int, bin_lo:float, bin_hi:float, n:int}>
     * }
     */
    public function aggregate(string $schema, string $scoreId, int $cohortDefinitionId, int $bins = 50): array
    {
        if (preg_match(self::SAFE_SCHEMA_REGEX, $schema) !== 1) {
            throw new \InvalidArgumentException("Unsafe schema: {$schema}");
        }
        if (preg_match('/^PGS\d+$/', $scoreId) !== 1) {
            throw new \InvalidArgumentException("Unsafe score_id: {$scoreId}");
        }
        $bins = max(10, min(200, $bins));

        // Summary + quintiles (one round-trip).
        $summaryRow = DB::connection('omop')->selectOne(
            "SELECT
                AVG(raw_score)::float AS mean,
                STDDEV(raw_score)::float AS stddev,
                MIN(raw_score)::float AS min,
                MAX(raw_score)::float AS max,
                percentile_cont(0.20) WITHIN GROUP (ORDER BY raw_score)::float AS q20,
                percentile_cont(0.40) WITHIN GROUP (ORDER BY raw_score)::float AS q40,
                percentile_cont(0.50) WITHIN GROUP (ORDER BY raw_score)::float AS median,
                percentile_cont(0.60) WITHIN GROUP (ORDER BY raw_score)::float AS q60,
                percentile_cont(0.80) WITHIN GROUP (ORDER BY raw_score)::float AS q80,
                percentile_cont(0.25) WITHIN GROUP (ORDER BY raw_score)::float AS iqr_q1,
                percentile_cont(0.75) WITHIN GROUP (ORDER BY raw_score)::float AS iqr_q3,
                COUNT(*)::int AS subject_count
              FROM {$schema}.prs_subject_scores
             WHERE score_id = ? AND cohort_definition_id = ?",
            [$scoreId, $cohortDefinitionId]
        );

        $subjectCount = (int) ($summaryRow->subject_count ?? 0);
        if ($subjectCount === 0) {
            return [
                'subject_count' => 0,
                'summary' => ['mean' => null, 'stddev' => null, 'min' => null, 'max' => null, 'median' => null, 'iqr_q1' => null, 'iqr_q3' => null],
                'quintiles' => ['q20' => null, 'q40' => null, 'q60' => null, 'q80' => null],
                'histogram' => [],
            ];
        }

        $histogramRows = DB::connection('omop')->select(
            "WITH
             bounds AS (
                 SELECT MIN(raw_score) AS lo, MAX(raw_score) AS hi
                   FROM {$schema}.prs_subject_scores
                  WHERE score_id = ? AND cohort_definition_id = ?
             ),
             binned AS (
                 SELECT width_bucket(raw_score, bounds.lo, bounds.hi, ?) AS bin, raw_score
                   FROM {$schema}.prs_subject_scores, bounds
                  WHERE score_id = ? AND cohort_definition_id = ?
             )
             SELECT LEAST(bin, ?)::int AS bin,
                    COUNT(*)::int AS n,
                    MIN(raw_score)::float AS bin_lo,
                    MAX(raw_score)::float AS bin_hi
               FROM binned
              GROUP BY LEAST(bin, ?)
              ORDER BY bin",
            [$scoreId, $cohortDefinitionId, $bins, $scoreId, $cohortDefinitionId, $bins, $bins]
        );

        $histogram = array_map(static function ($r): array {
            return [
                'bin' => (int) $r->bin,
                'bin_lo' => (float) $r->bin_lo,
                'bin_hi' => (float) $r->bin_hi,
                'n' => (int) $r->n,
            ];
        }, $histogramRows);

        return [
            'subject_count' => $subjectCount,
            'summary' => [
                'mean' => $summaryRow->mean !== null ? (float) $summaryRow->mean : null,
                'stddev' => $summaryRow->stddev !== null ? (float) $summaryRow->stddev : null,
                'min' => $summaryRow->min !== null ? (float) $summaryRow->min : null,
                'max' => $summaryRow->max !== null ? (float) $summaryRow->max : null,
                'median' => $summaryRow->median !== null ? (float) $summaryRow->median : null,
                'iqr_q1' => $summaryRow->iqr_q1 !== null ? (float) $summaryRow->iqr_q1 : null,
                'iqr_q3' => $summaryRow->iqr_q3 !== null ? (float) $summaryRow->iqr_q3 : null,
            ],
            'quintiles' => [
                'q20' => $summaryRow->q20 !== null ? (float) $summaryRow->q20 : null,
                'q40' => $summaryRow->q40 !== null ? (float) $summaryRow->q40 : null,
                'q60' => $summaryRow->q60 !== null ? (float) $summaryRow->q60 : null,
                'q80' => $summaryRow->q80 !== null ? (float) $summaryRow->q80 : null,
            ],
            'histogram' => $histogram,
        ];
    }
}
```

```php
<?php

declare(strict_types=1);

namespace App\Http\Controllers\Api\V1;

use App\Http\Controllers\Controller;
use App\Models\App\CohortDefinition;
use App\Models\App\PgsScore;
use App\Models\App\Source;
use App\Services\FinnGen\PrsAggregationService;
use Illuminate\Http\JsonResponse;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\DB;

final class CohortPrsController extends Controller
{
    public function __construct(
        private readonly PrsAggregationService $aggregator,
    ) {}

    /**
     * GET /api/v1/cohort-definitions/{id}/prs?bins=N
     *
     * Returns aggregated PRS distributions for every score computed against this cohort
     * across any source. Response contains NO per-subject raw scores (D-05 T-17-S2).
     */
    public function index(Request $request, int $id): JsonResponse
    {
        $cohort = CohortDefinition::findOrFail($id);
        $bins = max(10, min(200, (int) $request->query('bins', 50)));

        // Resolve the source schema from the cohort's source_id.
        // v1: single source per cohort. If no source_id, return empty.
        if ($cohort->source_id === null) {
            return response()->json(['scores' => []]);
        }
        /** @var Source|null $source */
        $source = Source::find($cohort->source_id);
        if ($source === null) {
            return response()->json(['scores' => []]);
        }
        $schema = strtolower((string) $source->source_key) . '_gwas_results';

        // Find which scores have been computed for this cohort (per DISTINCT score_id)
        // If the schema doesn't exist yet (source never had PRS computed), return empty.
        $schemaExists = DB::connection('omop')->selectOne(
            "SELECT EXISTS (SELECT 1 FROM information_schema.schemata WHERE schema_name = ?) AS e",
            [$schema]
        )->e;
        if (! $schemaExists) {
            return response()->json(['scores' => []]);
        }

        $scoreRows = DB::connection('omop')->select(
            "SELECT DISTINCT score_id, MAX(scored_at) AS scored_at
               FROM {$schema}.prs_subject_scores
              WHERE cohort_definition_id = ?
              GROUP BY score_id
              ORDER BY MAX(scored_at) DESC",
            [$id]
        );

        $out = [];
        foreach ($scoreRows as $row) {
            $scoreId = (string) $row->score_id;
            /** @var PgsScore|null $meta */
            $meta = PgsScore::find($scoreId);
            $agg = $this->aggregator->aggregate($schema, $scoreId, $id, $bins);
            $out[] = [
                'score_id' => $scoreId,
                'pgs_name' => $meta?->pgs_name,
                'trait_reported' => $meta?->trait_reported,
                'scored_at' => $row->scored_at,
                'subject_count' => $agg['subject_count'],
                'summary' => $agg['summary'],
                'quintiles' => $agg['quintiles'],
                'histogram' => $agg['histogram'],
            ];
        }

        return response()->json(['scores' => $out]);
    }
}
```

```php
public function download(
    \App\Http\Requests\FinnGen\DownloadPrsRequest $request,
    int $id,
    string $scoreId
): \Symfony\Component\HttpFoundation\StreamedResponse {
    $cohort = CohortDefinition::findOrFail($id);
    if ($cohort->source_id === null) {
        abort(404, 'Cohort has no source');
    }
    /** @var Source $source */
    $source = Source::findOrFail($cohort->source_id);
    $schema = strtolower((string) $source->source_key) . '_gwas_results';
    if (preg_match('/^[a-z][a-z0-9_]*$/', $schema) !== 1) {
        abort(500, "Unsafe schema: {$schema}");
    }

    $filename = "prs-{$scoreId}-cohort-{$id}.csv";

    return response()->streamDownload(function () use ($schema, $scoreId, $id): void {
        $out = fopen('php://output', 'w');
        if ($out === false) {
            throw new \RuntimeException('Failed to open php://output for streaming');
        }
```
