**Goal:** Give Abby relational understanding of clinical concepts — she traverses OMOP hierarchies, knows your CDM's strengths and gaps, and warns researchers before they build on sparse data.

**Architecture:** Three components: (1) KnowledgeGraphService — queries concept_ancestor and concept_relationship tables with Redis caching for hierarchy traversal, sibling/related concept discovery, and path finding, (2) DataProfileService — builds a living profile of the institution's CDM coverage (temporal range, domain density, vocabulary completeness) with proactive gap warnings, (3) Integration into Abby's chat pipeline via new live context tools and context assembler pieces. Uses existing PostgreSQL tables (no separate graph DB), Redis for hot path caching.

**Tech Stack:** Python 3.12 (FastAPI, SQLAlchemy, redis), PostgreSQL 16 (OMOP CDM v5.4 vocab schema), Redis 7

"""OMOP concept hierarchy traversal with Redis caching.

Queries concept_ancestor and concept_relationship tables to provide
relational understanding of clinical concepts. Results are cached in
Redis for fast repeated access.
"""
from __future__ import annotations

import json
import logging
from typing import Any

from sqlalchemy import text
from sqlalchemy.engine import Engine

from app.config import settings

logger = logging.getLogger(__name__)


class KnowledgeGraphService:
    """Traverses OMOP concept hierarchies using concept_ancestor and concept_relationship."""

    def __init__(
        self,
        engine: Engine,
        redis_client: Any = None,
        vocab_schema: str | None = None,
        cache_ttl: int | None = None,
        cache_prefix: str | None = None,
    ) -> None:
        self.engine = engine
        self.redis_client = redis_client
        self.vocab_schema = vocab_schema or settings.knowledge_vocab_schema
        self.cache_ttl = cache_ttl or settings.knowledge_cache_ttl
        self.cache_prefix = cache_prefix or settings.knowledge_cache_prefix

    def _cache_key(self, operation: str, concept_id: int, **kwargs: Any) -> str:
        extra = ":".join(f"{k}={v}" for k, v in sorted(kwargs.items()))
        return f"{self.cache_prefix}{operation}:{concept_id}:{extra}"

    def _get_cached(self, key: str) -> list[dict] | None:
        if not self.redis_client:
            return None
        try:
            data = self.redis_client.get(key)
            if data:
                return json.loads(data)
        except Exception:
            logger.debug("Cache miss for %s", key)
        return None

    def _set_cached(self, key: str, data: list[dict]) -> None:
        if not self.redis_client:
            return
        try:
            self.redis_client.setex(key, self.cache_ttl, json.dumps(data))
        except Exception:
            logger.debug("Cache set failed for %s", key)

    def get_concept(self, concept_id: int) -> dict | None:
        """Get details for a single concept."""
        try:
            with self.engine.connect() as conn:
                row = conn.execute(
                    text(f"""
                        SELECT concept_id, concept_name, domain_id,
                               vocabulary_id, concept_class_id, standard_concept
                        FROM {self.vocab_schema}.concept
                        WHERE concept_id = :cid
                    """),
                    {"cid": concept_id},
                ).fetchone()
                if row:
                    return {
                        "concept_id": row[0],
                        "concept_name": row[1],
                        "domain_id": row[2],
                        "vocabulary_id": row[3],
                        "concept_class_id": row[4],
                        "standard_concept": row[5],
                    }
        except Exception:
            logger.exception("Failed to get concept %d", concept_id)
        return None

    def get_ancestors(
        self,
        concept_id: int,
        max_levels: int | None = None,
    ) -> list[dict]:
        """Get ancestor concepts in the hierarchy."""
        max_levels = max_levels or settings.knowledge_max_traversal_depth
        cache_key = self._cache_key("ancestors", concept_id, max_levels=max_levels)
        cached = self._get_cached(cache_key)
        if cached is not None:
            return cached

        try:
            with self.engine.connect() as conn:
                rows = conn.execute(
                    text(f"""
                        SELECT c.concept_id, c.concept_name, c.domain_id,
                               ca.min_levels_of_separation, ca.max_levels_of_separation
                        FROM {self.vocab_schema}.concept_ancestor ca
                        JOIN {self.vocab_schema}.concept c ON c.concept_id = ca.ancestor_concept_id
                        WHERE ca.descendant_concept_id = :cid
                          AND ca.ancestor_concept_id != :cid
                          AND ca.min_levels_of_separation <= :max_levels
                        ORDER BY ca.min_levels_of_separation ASC
                    """),
                    {"cid": concept_id, "max_levels": max_levels},
                ).fetchall()

            result = [
                {
                    "concept_id": row[0],
                    "concept_name": row[1],
                    "domain_id": row[2],
                    "min_separation": row[3],
                    "max_separation": row[4],
                }
                for row in rows
            ]
            self._set_cached(cache_key, result)
            return result
        except Exception:
            logger.exception("Failed to get ancestors for concept %d", concept_id)
            return []

    def get_descendants(
        self,
        concept_id: int,
        max_levels: int | None = None,
    ) -> list[dict]:
        """Get descendant concepts in the hierarchy."""
        max_levels = max_levels or settings.knowledge_max_traversal_depth
        cache_key = self._cache_key("descendants", concept_id, max_levels=max_levels)
        cached = self._get_cached(cache_key)
        if cached is not None:
            return cached

        try:
            with self.engine.connect() as conn:
                rows = conn.execute(
                    text(f"""
                        SELECT c.concept_id, c.concept_name, c.domain_id,
                               ca.min_levels_of_separation, ca.max_levels_of_separation
                        FROM {self.vocab_schema}.concept_ancestor ca
                        JOIN {self.vocab_schema}.concept c ON c.concept_id = ca.descendant_concept_id
                        WHERE ca.ancestor_concept_id = :cid
                          AND ca.descendant_concept_id != :cid
                          AND ca.min_levels_of_separation <= :max_levels
                        ORDER BY ca.min_levels_of_separation ASC
                        LIMIT 50
                    """),
                    {"cid": concept_id, "max_levels": max_levels},
                ).fetchall()

            result = [
                {
                    "concept_id": row[0],
                    "concept_name": row[1],
                    "domain_id": row[2],
                    "min_separation": row[3],
                    "max_separation": row[4],
                }
                for row in rows
            ]
            self._set_cached(cache_key, result)
            return result
        except Exception:
            logger.exception("Failed to get descendants for concept %d", concept_id)
            return []

    def get_siblings(self, concept_id: int) -> list[dict]:
        """Get sibling concepts (same immediate parent)."""
        cache_key = self._cache_key("siblings", concept_id)
        cached = self._get_cached(cache_key)
        if cached is not None:
            return cached

        try:
            # Step 1: Find immediate parent(s)
            parents = self.get_ancestors(concept_id, max_levels=1)
            if not parents:
                return []

            # Step 2: Get children of the first parent
            parent_id = parents[0]["concept_id"]
            children = self.get_descendants(parent_id, max_levels=1)

            # Step 3: Exclude the original concept
            result = [c for c in children if c["concept_id"] != concept_id]
            self._set_cached(cache_key, result)
            return result
        except Exception:
            logger.exception("Failed to get siblings for concept %d", concept_id)
            return []

    def find_related(
        self,
        concept_id: int,
        relationship_types: list[str] | None = None,
    ) -> list[dict]:
        """Find related concepts via concept_relationship."""
        cache_key = self._cache_key("related", concept_id,
                                     types=",".join(relationship_types) if relationship_types else "all")
        cached = self._get_cached(cache_key)
        if cached is not None:
            return cached

        try:
            type_filter = ""
            params: dict[str, Any] = {"cid": concept_id}
            if relationship_types:
                placeholders = ", ".join(f":rt{i}" for i in range(len(relationship_types)))
                type_filter = f"AND cr.relationship_id IN ({placeholders})"
                for i, rt in enumerate(relationship_types):
                    params[f"rt{i}"] = rt

            with self.engine.connect() as conn:
                rows = conn.execute(
                    text(f"""
                        SELECT c.concept_id, c.concept_name, c.domain_id,
                               cr.relationship_id, r.relationship_name
                        FROM {self.vocab_schema}.concept_relationship cr
                        JOIN {self.vocab_schema}.concept c ON c.concept_id = cr.concept_id_2
                        JOIN {self.vocab_schema}.relationship r ON r.relationship_id = cr.relationship_id
                        WHERE cr.concept_id_1 = :cid
                          AND cr.invalid_reason IS NULL
                          {type_filter}
                        LIMIT 30
                    """),
                    params,
                ).fetchall()

            result = [
                {
                    "concept_id": row[0],
                    "concept_name": row[1],
                    "domain_id": row[2],
                    "relationship_id": row[3],
                    "relationship_name": row[4],
                }
                for row in rows
            ]
            self._set_cached(cache_key, result)
            return result
        except Exception:
            logger.exception("Failed to find related concepts for %d", concept_id)
            return []

    def format_hierarchy(self, concepts: list[dict], direction: str = "ancestors") -> str:
        """Format hierarchy results for prompt injection."""
        if not concepts:
            return ""
        lines = [f"Concept hierarchy ({direction}):"]
        for c in concepts:
            depth = c.get("min_separation", "?")
            lines.append(f"  {'  ' * (int(depth) if isinstance(depth, int) else 0)}↳ {c['concept_name']} "
                        f"({c.get('domain_id', '')}, depth={depth})")
        return "\n".join(lines)

    def format_related(self, concepts: list[dict]) -> str:
        """Format related concepts for prompt injection."""
        if not concepts:
            return ""
        lines = ["Related concepts:"]
        for c in concepts:
            lines.append(f"  → {c['concept_name']} ({c.get('relationship_name', '')})")
        return "\n".join(lines)

```python
"""CDM data profile — coverage analysis, gap detection, and quality warnings.

Builds a living understanding of the institution's CDM: which domains have
strong data, which are sparse, temporal coverage range, and known data
quality issues. Surfaces proactive warnings when researchers explore
domains with limited data.
"""
from __future__ import annotations

import json
import logging
from dataclasses import dataclass
from datetime import date, datetime
from typing import Any

from sqlalchemy import text
from sqlalchemy.engine import Engine

from app.config import settings

logger = logging.getLogger(__name__)

# Minimum records per 1000 patients to consider a domain well-populated
DOMAIN_DENSITY_THRESHOLD = 10  # records per patient
SPARSE_DOMAIN_THRESHOLD = 1  # records per patient = sparse


@dataclass
class DataGapWarning:
    gap_type: str  # sparse_domain, temporal, quality
    domain: str
    severity: str  # info, warning, critical
    message: str


class DataProfileService:
    """Profiles CDM data coverage and detects gaps."""

    def __init__(
        self,
        engine: Engine,
        redis_client: Any = None,
        cdm_schema: str | None = None,
        cache_ttl: int = 3600,
    ) -> None:
        self.engine = engine
        self.redis_client = redis_client
        self.cdm_schema = cdm_schema or settings.knowledge_cdm_schema
        self.cache_ttl = cache_ttl

    def get_person_count(self) -> int:
        """Get total person count in the CDM."""
        try:
            with self.engine.connect() as conn:
                row = conn.execute(
                    text(f"SELECT COUNT(*) FROM {self.cdm_schema}.person"),
                ).fetchone()
                return int(row[0]) if row else 0
        except Exception:
            logger.exception("Failed to get person count")
            return 0

    def get_temporal_coverage(self) -> dict[str, str]:
        """Get the temporal range of observation periods."""
        try:
            with self.engine.connect() as conn:
                row = conn.execute(
                    text(f"""
                        SELECT MIN(observation_period_start_date),
                               MAX(observation_period_end_date)
                        FROM {self.cdm_schema}.observation_period
                    """),
                ).fetchone()
                if row and row[0] and row[1]:
                    return {
                        "start_date": str(row[0]),
                        "end_date": str(row[1]),
                    }
        except Exception:
            logger.exception("Failed to get temporal coverage")
        return {"start_date": "unknown", "end_date": "unknown"}

    def get_domain_density(self) -> list[dict]:
        """Get record counts per clinical domain, ordered by density."""
        domain_tables = {
            "Condition": "condition_occurrence",
            "Drug": "drug_exposure",
            "Procedure": "procedure_occurrence",
            "Measurement": "measurement",
            "Observation": "observation",
            "Visit": "visit_occurrence",
            "Device": "device_exposure",
        }
        results = []
        try:
            with self.engine.connect() as conn:
                for domain, table in domain_tables.items():
                    try:
                        row = conn.execute(
                            text(f"SELECT COUNT(*) FROM {self.cdm_schema}.{table}"),
                        ).fetchone()
                        count = int(row[0]) if row else 0
                        results.append({"domain": domain, "record_count": count})
                    except Exception:
                        results.append({"domain": domain, "record_count": 0})
        except Exception:
            logger.exception("Failed to get domain density")

        results.sort(key=lambda x: x["record_count"], reverse=True)
        return results

    def detect_data_gaps(self) -> list[DataGapWarning]:
        """Analyze CDM for data quality gaps and sparse domains."""
        warnings: list[DataGapWarning] = []
        person_count = self.get_person_count()

        if person_count == 0:
            warnings.append(DataGapWarning(
                gap_type="empty_cdm", domain="all",
                severity="critical",
                message="CDM has no patient data loaded. All clinical queries will return empty results.",
            ))
            return warnings

        # Check domain density
        domains = self.get_domain_density()
        for d in domains:
            records_per_patient = d["record_count"] / person_count if person_count > 0 else 0
            if records_per_patient < SPARSE_DOMAIN_THRESHOLD and d["record_count"] > 0:
                warnings.append(DataGapWarning(
                    gap_type="sparse_domain", domain=d["domain"],
                    severity="warning",
                    message=f"{d['domain']} domain has very sparse data "
                            f"({d['record_count']:,} records for {person_count:,} patients, "
                            f"{records_per_patient:.1f} per patient)",
                ))
            elif d["record_count"] == 0:
                warnings.append(DataGapWarning(
                    gap_type="sparse_domain", domain=d["domain"],
                    severity="warning",
                    message=f"{d['domain']} domain has no data",
                ))

        # Check temporal coverage
        temporal = self.get_temporal_coverage()
        if temporal["start_date"] != "unknown" and temporal["end_date"] != "unknown":
            try:
                start = datetime.strptime(temporal["start_date"], "%Y-%m-%d").date()
                end = datetime.strptime(temporal["end_date"], "%Y-%m-%d").date()
                span_years = (end - start).days / 365.25
                if span_years < 3:
                    warnings.append(DataGapWarning(
                        gap_type="temporal", domain="all",
                        severity="warning",
                        message=f"CDM has limited temporal coverage: {temporal['start_date']} to "
                                f"{temporal['end_date']} ({span_years:.1f} years). "
                                f"Longitudinal studies may have insufficient follow-up.",
                    ))
            except (ValueError, TypeError):
                pass

        return warnings

    def format_warnings(self, warnings: list[DataGapWarning]) -> str:
        """Format warnings for prompt injection as safety-critical context."""
        if not warnings:
            return ""
        lines = ["DATA QUALITY WARNINGS:"]
        for w in warnings:
            icon = "⚠️" if w.severity == "warning" else "🔴" if w.severity == "critical" else "ℹ️"
            lines.append(f"  {icon} [{w.domain}] {w.message}")
        return "\n".join(lines)

    def get_profile_summary(self) -> dict:
        """Get a complete CDM profile summary."""
        person_count = self.get_person_count()
        temporal = self.get_temporal_coverage()
        domains = self.get_domain_density()
        warnings = self.detect_data_gaps()

        return {
            "person_count": person_count,
            "temporal_coverage": temporal,
            "domain_density": domains,
            "warnings": [
                {"type": w.gap_type, "domain": w.domain,
                 "severity": w.severity, "message": w.message}
                for w in warnings
            ],
        }
```
