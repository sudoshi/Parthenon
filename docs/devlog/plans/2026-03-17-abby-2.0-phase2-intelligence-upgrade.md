# Abby 2.0 Phase 2: Intelligence Upgrade — Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Give Abby a hybrid brain — MedGemma handles 70% of requests locally (fast, free), Claude API handles 30% requiring deep reasoning (cloud, powerful), with strict PHI protection and cost controls ensuring no clinical data leaves the network and spending stays within budget.

**Architecture:** Three subsystems: (1) Two-stage model router — deterministic rule-based pre-filter + bootstrap rule engine (classifier deferred until 500+ labeled samples collected), (2) PHI sanitization layer — architectural allowlist boundary + regex/spaCy NER defense-in-depth with circuit breaker, (3) Cost control system — `abby_cloud_usage` table, budget tracking, circuit breaker at 95% that falls back to local-only. All three integrate into the existing chat pipeline via the context assembly pipeline from Phase 1.

**Tech Stack:** Python 3.12 (FastAPI, anthropic SDK, spaCy), PHP 8.4 (Laravel 11), React 19 (TypeScript), PostgreSQL 16

```python
"""PHI sanitizer — detects and blocks Protected Health Information in cloud-bound prompts.

Combines regex patterns for structured identifiers (SSN, MRN, phone, email, DOB)
with optional spaCy NER for name detection. Designed as defense-in-depth behind
the architectural allowlist boundary.

Compliance target: HIPAA Safe Harbor method.
"""
from __future__ import annotations

import hashlib
import logging
import re
from dataclasses import dataclass, field
from typing import Any

logger = logging.getLogger(__name__)


@dataclass
class PHIFinding:
    pattern_type: str  # ssn, mrn, phone, email, dob, person_name
    matched_text: str
    start: int
    end: int


@dataclass
class SanitizationResult:
    phi_detected: bool
    redacted_text: str
    redaction_count: int
    findings: list[PHIFinding] = field(default_factory=list)
    original_hash: str = ""

    @property
    def is_safe(self) -> bool:
        return not self.phi_detected


# Regex patterns for PHI detection
# Each pattern is a tuple of (name, compiled_regex, context_validator)
PHI_PATTERNS: list[tuple[str, re.Pattern, Any]] = [
    # SSN: 123-45-6789 or 123456789
    ("ssn", re.compile(r'\b\d{3}-\d{2}-\d{4}\b'), None),
    ("ssn", re.compile(r'\bSSN\s*[:=]?\s*\d{9}\b', re.IGNORECASE), None),

    # MRN: 6-10 digit number preceded by MRN/medical record context
    ("mrn", re.compile(
        r'(?:MRN|medical\s+record(?:\s+number)?|patient\s+(?:id|number))\s*[:=]?\s*(\d{6,10})\b',
        re.IGNORECASE,
    ), None),

    # Phone: (555) 123-4567 or 555-123-4567 or 555.123.4567
    ("phone", re.compile(
        r'(?:call|phone|tel|contact|reach)\s+.*?\(?\d{3}\)?[-.\s]\d{3}[-.\s]\d{4}\b',
        re.IGNORECASE,
    ), None),
    ("phone", re.compile(r'\(\d{3}\)\s*\d{3}[-.\s]\d{4}\b'), None),

    # Email
    ("email", re.compile(
        r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b'
    ), None),

    # DOB: preceded by DOB/born/birth context + date pattern
    ("dob", re.compile(
        r'(?:DOB|date\s+of\s+birth|born|birthday)\s*[:=]?\s*'
        r'(?:\d{1,2}[/\-]\d{1,2}[/\-]\d{2,4})',
        re.IGNORECASE,
    ), None),
]

# Terms that indicate a number is a clinical identifier, not PHI
CLINICAL_CONTEXT_TERMS = {
    "concept", "concept_id", "snomed", "icd", "rxnorm", "loinc",
    "analysis", "analysis_id", "cohort", "cohort_id", "achilles",
    "vocabulary", "domain", "person_count", "count", "total",
}


class PHISanitizer:
    """Detects and optionally redacts PHI from text before cloud API calls."""

    def __init__(self, use_ner: bool = True) -> None:
        self.use_ner = use_ner
        self._nlp = None

    def _get_nlp(self):
        """Lazy-load spaCy model."""
        if self._nlp is None and self.use_ner:
            try:
                import spacy
                self._nlp = spacy.load("en_core_web_sm")
            except Exception:
                logger.warning("spaCy model not available, falling back to regex-only PHI detection")
                self.use_ner = False
        return self._nlp

    def scan(self, text: str) -> SanitizationResult:
        """Scan text for PHI. Returns detection result with optional redaction."""
        if not text:
            return SanitizationResult(
                phi_detected=False,
                redacted_text=text,
                redaction_count=0,
                original_hash=hashlib.sha256(text.encode()).hexdigest(),
            )

        findings: list[PHIFinding] = []
        original_hash = hashlib.sha256(text.encode()).hexdigest()

        # Stage 1: Regex-based detection
        for pattern_name, pattern, validator in PHI_PATTERNS:
            for match in pattern.finditer(text):
                # Check if the match is in a clinical context (false positive guard)
                start = max(0, match.start() - 50)
                context_window = text[start:match.start()].lower()
                if any(term in context_window for term in CLINICAL_CONTEXT_TERMS):
                    continue

                findings.append(PHIFinding(
                    pattern_type=pattern_name,
                    matched_text=match.group(),
                    start=match.start(),
                    end=match.end(),
                ))

        # Stage 2: NER-based name detection (if enabled)
        if self.use_ner:
            nlp = self._get_nlp()
            if nlp:
                try:
                    doc = nlp(text)
                    for ent in doc.ents:
                        if ent.label_ == "PERSON":
                            findings.append(PHIFinding(
                                pattern_type="person_name",
                                matched_text=ent.text,
                                start=ent.start_char,
                                end=ent.end_char,
                            ))
                except Exception:
                    logger.warning("NER PHI detection failed, using regex results only")

        # Build redacted text
        redacted_text = text
        # Sort findings by position (reverse) to avoid offset issues during replacement
        sorted_findings = sorted(findings, key=lambda f: f.start, reverse=True)
        for finding in sorted_findings:
            redacted_text = (
                redacted_text[:finding.start]
                + "[REDACTED]"
                + redacted_text[finding.end:]
            )

        # Deduplicate overlapping findings
        unique_findings = _deduplicate_findings(findings)

        return SanitizationResult(
            phi_detected=len(unique_findings) > 0,
            redacted_text=redacted_text,
            redaction_count=len(unique_findings),
            findings=unique_findings,
            original_hash=original_hash,
        )


def _deduplicate_findings(findings: list[PHIFinding]) -> list[PHIFinding]:
    """Remove overlapping findings, keeping the longer match."""
    if not findings:
        return []
    sorted_f = sorted(findings, key=lambda f: (f.start, -(f.end - f.start)))
    result = [sorted_f[0]]
    for f in sorted_f[1:]:
        prev = result[-1]
        if f.start >= prev.end:
            result.append(f)
    return result
```

```python
"""Cloud safety filter — allowlist-based context filtering for cloud API calls.

Primary defense: architectural boundary. Only approved data sources can be included
in prompts sent to cloud LLMs. Individual-level clinical data never leaves the network.

Compliance target: HIPAA Safe Harbor method.
"""
from __future__ import annotations

import logging
import re
from typing import TYPE_CHECKING

if TYPE_CHECKING:
    from app.memory.context_assembler import ContextPiece, ContextTier

logger = logging.getLogger(__name__)

# Sources that are ALWAYS safe for cloud
CLOUD_SAFE_SOURCES = {
    "help_docs", "faq_shared", "user_profile", "page_context",
    "achilles_stats", "dqd_summary", "cdm_summary",
    "concept_sets", "cohort_definitions", "analyses",
    "vocabulary", "clinical_reference", "ohdsi_papers",
}

# Sources that are NEVER safe for cloud (individual-level clinical data)
CLOUD_BLOCKED_SOURCES = {
    "cdm.person", "cdm.visit_occurrence", "cdm.condition_occurrence",
    "cdm.drug_exposure", "cdm.procedure_occurrence", "cdm.measurement",
    "cdm.observation", "cdm.death", "cdm.note", "cdm.note_nlp",
    "cdm.specimen", "cdm.device_exposure", "cdm.payer_plan_period",
    "cdm.cost", "cdm.observation_period",
}

# Patterns indicating individual-level data in content
INDIVIDUAL_DATA_PATTERNS = [
    re.compile(r'\bperson_id\s*[:=]\s*\d+', re.IGNORECASE),
    re.compile(r'\bvisit_occurrence_id\s*[:=]\s*\d+', re.IGNORECASE),
    re.compile(r'\bbirth_datetime\s*[:=]', re.IGNORECASE),
    re.compile(r'\bpatient_id\s*[:=]\s*\d+', re.IGNORECASE),
]


class CloudSafetyFilter:
    """Filters context pieces to ensure only cloud-safe data is sent to external APIs."""

    def is_cloud_safe(self, piece: "ContextPiece") -> bool:
        """Check if a context piece is safe to include in cloud API calls."""
        from app.memory.context_assembler import ContextTier

        # Tier-based allowlist: Working, Page, Episodic, Semantic, Institutional always safe
        always_safe_tiers = {
            ContextTier.WORKING, ContextTier.PAGE, ContextTier.EPISODIC,
            ContextTier.SEMANTIC, ContextTier.INSTITUTIONAL,
        }
        if piece.tier in always_safe_tiers:
            return True

        # Live context needs source-level checking
        if piece.tier == ContextTier.LIVE:
            # Check source against blocked list
            if piece.source in CLOUD_BLOCKED_SOURCES:
                logger.warning("Blocked cloud-unsafe source: %s", piece.source)
                return False

            # Check source against safe list
            if piece.source in CLOUD_SAFE_SOURCES:
                return True

            # Unknown source: check content for individual-level data patterns
            for pattern in INDIVIDUAL_DATA_PATTERNS:
                if pattern.search(piece.content):
                    logger.warning(
                        "Blocked cloud-unsafe content pattern in source=%s",
                        piece.source,
                    )
                    return False

            # Unknown source with no individual data patterns: allow with warning
            logger.info("Allowing unknown source for cloud: %s", piece.source)
            return True

        return True

    def filter_for_cloud(self, pieces: list["ContextPiece"]) -> list["ContextPiece"]:
        """Filter a list of context pieces, removing cloud-unsafe ones."""
        return [p for p in pieces if self.is_cloud_safe(p)]
```

```python
"""Two-stage model router — deterministic rules + bootstrap scoring.

Stage 1: Deterministic rule-based pre-filter (zero cost, <1ms)
Stage 2: Bootstrap heuristic scoring (until 500+ labeled samples for classifier)

The router errs toward Claude when uncertain — better to over-spend than
under-deliver on a clinical research question.
"""
from __future__ import annotations

import re
import logging
from dataclasses import dataclass

logger = logging.getLogger(__name__)


@dataclass(frozen=True)
class RoutingDecision:
    model: str       # "local" or "claude"
    stage: int       # 1 (deterministic) or 2 (scoring)
    reason: str      # Human-readable reason
    confidence: float  # 0.0-1.0


# Stage 1: Deterministic cloud triggers
CLOUD_ACTION_WORDS = re.compile(
    r'\b(create|build|run|modify|delete|design|construct|generate|execute|schedule)\b',
    re.IGNORECASE,
)

# Stage 1: Deterministic local triggers
LOCAL_GREETINGS = re.compile(
    r'^(hello|hi|hey|good\s+(morning|afternoon|evening)|thanks?|thank\s+you|bye|goodbye)\b',
    re.IGNORECASE,
)

LOCAL_SIMPLE_LOOKUP = re.compile(
    r'^(what\s+is|how\s+many|show\s+me|list|count)\b',
    re.IGNORECASE,
)

# Stage 2: Complexity indicators (boost toward cloud)
COMPLEXITY_INDICATORS = [
    re.compile(r'\b(interpret|analyze|critique|review|compare|evaluate|assess)\b', re.IGNORECASE),
    re.compile(r'\b(best\s+practice|methodology|bias|confounding|causal)\b', re.IGNORECASE),
    re.compile(r'\b(immortal\s+time|propensity|hazard\s+ratio|kaplan|cox)\b', re.IGNORECASE),
    re.compile(r'\b(explain\s+why|what\s+does\s+this\s+mean|implications)\b', re.IGNORECASE),
    re.compile(r'\b(multi.?step|workflow|pipeline|sequence|chain)\b', re.IGNORECASE),
    re.compile(r'\b(SQL|query|join|subquery)\b', re.IGNORECASE),
]

# Stage 2: Simplicity indicators (boost toward local)
SIMPLICITY_INDICATORS = [
    re.compile(r'^what\s+is\s+\w+\s*\??\s*$', re.IGNORECASE),
    re.compile(r'^(yes|no|ok|sure|got\s+it)\b', re.IGNORECASE),
    re.compile(r'^show\s+me\s+', re.IGNORECASE),
    re.compile(r'^list\s+', re.IGNORECASE),
]

# Message length threshold for complexity
LONG_MESSAGE_THRESHOLD = 200


class RuleRouter:
    """Two-stage model router with deterministic rules and bootstrap scoring."""

    def route(
        self,
        message: str,
        budget_exhausted: bool = False,
    ) -> RoutingDecision:
        """Route a message to the appropriate model."""

        # Budget override: if cloud budget is exhausted, always use local
        if budget_exhausted:
            return RoutingDecision(
                model="local",
                stage=0,
                reason="budget_exhausted",
                confidence=1.0,
            )

        # Stage 1: Deterministic rules
        stage1_result = self._stage1_deterministic(message)
        if stage1_result is not None:
            return stage1_result

        # Stage 2: Bootstrap heuristic scoring
        return self._stage2_scoring(message)

    def _stage1_deterministic(self, message: str) -> RoutingDecision | None:
        """Apply deterministic rules. Returns None if no rule matches."""
        msg_stripped = message.strip()

        # Cloud: action words
        if CLOUD_ACTION_WORDS.search(msg_stripped):
            return RoutingDecision(
                model="claude",
                stage=1,
                reason="action_word_detected",
                confidence=0.9,
            )

        # Cloud: long message (likely complex)
        if len(msg_stripped) > LONG_MESSAGE_THRESHOLD:
            # Check for multiple clauses (commas, semicolons, "and", "but", "or")
            clause_markers = len(re.findall(r'[,;]|\band\b|\bbut\b|\bor\b', msg_stripped))
            if clause_markers >= 2:
                return RoutingDecision(
                    model="claude",
                    stage=1,
                    reason="long_complex_message",
                    confidence=0.85,
                )

        # Local: greetings
        if LOCAL_GREETINGS.match(msg_stripped):
            return RoutingDecision(
                model="local",
                stage=1,
                reason="greeting_detected",
                confidence=0.95,
            )

        # Local: simple lookup (short question starting with what/how/show/list)
        if LOCAL_SIMPLE_LOOKUP.match(msg_stripped) and len(msg_stripped) < 80:
            return RoutingDecision(
                model="local",
                stage=1,
                reason="simple_lookup",
                confidence=0.8,
            )

        return None  # No deterministic match → Stage 2

    def _stage2_scoring(self, message: str) -> RoutingDecision:
        """Bootstrap heuristic scoring until classifier is trained."""
        cloud_score = 0.0
        local_score = 0.0

        # Complexity indicators boost cloud
        for pattern in COMPLEXITY_INDICATORS:
            if pattern.search(message):
                cloud_score += 0.2

        # Simplicity indicators boost local
        for pattern in SIMPLICITY_INDICATORS:
            if pattern.search(message):
                local_score += 0.3

        # Message length factor
        msg_len = len(message)
        if msg_len > 150:
            cloud_score += 0.15
        elif msg_len < 50:
            local_score += 0.15

        # Question mark count (more = more complex)
        question_marks = message.count("?")
        if question_marks > 1:
            cloud_score += 0.1

        # Default: err toward cloud when uncertain (per spec)
        if cloud_score == local_score:
            cloud_score += 0.05  # Slight bias toward capability

        if cloud_score > local_score:
            confidence = min(0.5 + cloud_score, 0.95)
            return RoutingDecision(
                model="claude",
                stage=2,
                reason="complexity_scoring",
                confidence=confidence,
            )

        confidence = min(0.5 + local_score, 0.95)
        return RoutingDecision(
            model="local",
            stage=2,
            reason="simplicity_scoring",
            confidence=confidence,
        )
```

```python
"""Claude API client wrapper with cost estimation and audit logging.

Wraps the Anthropic Python SDK for Abby's hybrid LLM architecture.
All calls are logged for cost tracking and compliance audit.
"""
from __future__ import annotations

import hashlib
import logging
import time
from dataclasses import dataclass
from typing import Any

import anthropic

from app.config import settings

logger = logging.getLogger(__name__)

# Claude Sonnet 4 pricing (per million tokens) — update when model changes
PRICING = {
    "claude-sonnet-4-20250514": {"input": 3.00, "output": 15.00},
    "claude-opus-4-6-20250514": {"input": 15.00, "output": 75.00},
}
DEFAULT_PRICING = {"input": 3.00, "output": 15.00}


@dataclass
class ClaudeResponse:
    reply: str
    tokens_in: int
    tokens_out: int
    cost_usd: float
    model: str
    latency_ms: int
    request_hash: str


class ClaudeClient:
    """Wrapper around the Anthropic Python SDK for Abby chat."""

    def __init__(
        self,
        api_key: str | None = None,
        model: str | None = None,
        max_tokens: int | None = None,
        timeout: int | None = None,
    ) -> None:
        self.api_key = api_key or settings.claude_api_key
        if not self.api_key:
            raise ValueError("Claude API key is required. Set CLAUDE_API_KEY env var.")
        self.model = model or settings.claude_model
        self.max_tokens = max_tokens or settings.claude_max_tokens
        self.timeout = timeout or settings.claude_timeout
        self._client = anthropic.Anthropic(api_key=self.api_key)

    def chat(
        self,
        system_prompt: str,
        message: str,
        history: list[dict[str, str]] | None = None,
        temperature: float = 0.15,
    ) -> ClaudeResponse:
        """Send a message to Claude and return the response."""
        messages: list[dict[str, Any]] = []

        # Add conversation history
        if history:
            for msg in history:
                messages.append({
                    "role": msg["role"],
                    "content": msg["content"],
                })

        # Add current message
        messages.append({"role": "user", "content": message})

        # Hash the request for audit trail
        request_hash = hashlib.sha256(
            (system_prompt + message).encode()
        ).hexdigest()

        start_time = time.monotonic()
        try:
            response = self._client.messages.create(
                model=self.model,
                max_tokens=self.max_tokens,
                system=system_prompt,
                messages=messages,
                temperature=temperature,
                timeout=self.timeout,
            )
        except anthropic.APIStatusError as e:
            logger.error("Claude API error: %s", e)
            raise
        except anthropic.APIConnectionError as e:
            logger.error("Claude API connection error: %s", e)
            raise

        latency_ms = int((time.monotonic() - start_time) * 1000)

        tokens_in = response.usage.input_tokens
        tokens_out = response.usage.output_tokens
        cost = self.estimate_cost(tokens_in, tokens_out)

        reply = response.content[0].text if response.content else ""

        logger.info(
            "Claude call: model=%s tokens_in=%d tokens_out=%d cost=$%.4f latency=%dms",
            self.model, tokens_in, tokens_out, cost, latency_ms,
        )

        return ClaudeResponse(
            reply=reply,
            tokens_in=tokens_in,
            tokens_out=tokens_out,
            cost_usd=cost,
            model=response.model,
            latency_ms=latency_ms,
            request_hash=request_hash,
        )

    def estimate_cost(self, tokens_in: int, tokens_out: int) -> float:
        """Estimate cost in USD for a given token count."""
        pricing = PRICING.get(self.model, DEFAULT_PRICING)
        cost_in = (tokens_in / 1_000_000) * pricing["input"]
        cost_out = (tokens_out / 1_000_000) * pricing["output"]
        return round(cost_in + cost_out, 6)
```

```python
"""Cost tracker — cloud API usage tracking and budget enforcement.

Records every Claude API call to the abby_cloud_usage table and enforces
monthly budget limits via a circuit breaker pattern.
"""
from __future__ import annotations

import logging
from datetime import datetime

from sqlalchemy import text
from sqlalchemy.engine import Engine

from app.config import settings

logger = logging.getLogger(__name__)


class CostTracker:
    """Tracks cloud API costs and enforces budget limits."""

    def __init__(
        self,
        engine: Engine,
        monthly_budget: float | None = None,
        alert_threshold: float | None = None,
        cutoff_threshold: float | None = None,
    ) -> None:
        self.engine = engine
        self.monthly_budget = monthly_budget or settings.cloud_monthly_budget_usd
        self.alert_threshold = alert_threshold or 0.80
        self.cutoff_threshold = cutoff_threshold or settings.cloud_budget_cutoff_threshold
        self.alert_thresholds = settings.cloud_budget_alert_thresholds

    def record_usage(
        self,
        user_id: int | None,
        tokens_in: int,
        tokens_out: int,
        cost_usd: float,
        model: str,
        request_hash: str = "",
        redaction_count: int = 0,
        route_reason: str = "",
        department: str = "",
    ) -> None:
        """Record a cloud API call to the database."""
        try:
            with self.engine.connect() as conn:
                conn.execute(
                    text("""
                        INSERT INTO app.abby_cloud_usage
                            (user_id, department, tokens_in, tokens_out, cost_usd, model,
                             request_hash, sanitizer_redaction_count, route_reason, created_at)
                        VALUES
                            (:user_id, :department, :tokens_in, :tokens_out, :cost_usd, :model,
                             :request_hash, :redaction_count, :route_reason, NOW())
                    """),
                    {
                        "user_id": user_id,
                        "department": department or None,
                        "tokens_in": tokens_in,
                        "tokens_out": tokens_out,
                        "cost_usd": cost_usd,
                        "model": model,
                        "request_hash": request_hash,
                        "redaction_count": redaction_count,
                        "route_reason": route_reason,
                    },
                )
                conn.commit()
        except Exception:
            logger.exception("Failed to record cloud usage")

    def get_monthly_spend(self) -> float:
        """Get total cloud API spend for the current month."""
        try:
            now = datetime.now()
            month_start = now.replace(day=1, hour=0, minute=0, second=0, microsecond=0)

            with self.engine.connect() as conn:
                row = conn.execute(
                    text("""
                        SELECT COALESCE(SUM(cost_usd), 0)
                        FROM app.abby_cloud_usage
                        WHERE created_at >= :month_start
                    """),
                    {"month_start": month_start},
                ).fetchone()
                return float(row[0]) if row else 0.0
        except Exception:
            logger.exception("Failed to get monthly spend")
            return 0.0

    def is_budget_exhausted(self) -> bool:
        """Check if cloud spending has exceeded the cutoff threshold."""
        spend = self.get_monthly_spend()
        return spend >= (self.monthly_budget * self.cutoff_threshold)

    def should_alert(self) -> bool:
        """Check if cloud spending has crossed any alert threshold."""
        spend = self.get_monthly_spend()
        return any(spend >= (self.monthly_budget * t) for t in self.alert_thresholds)

    def get_triggered_alerts(self) -> list[float]:
        """Return list of alert thresholds that have been crossed."""
        spend = self.get_monthly_spend()
        return [t for t in sorted(self.alert_thresholds) if spend >= (self.monthly_budget * t)]

    def get_budget_status(self) -> dict:
        """Get a summary of current budget usage."""
        spend = self.get_monthly_spend()
        return {
            "monthly_budget": self.monthly_budget,
            "current_spend": spend,
            "remaining": max(0, self.monthly_budget - spend),
            "utilization_pct": round((spend / self.monthly_budget) * 100, 1) if self.monthly_budget > 0 else 0,
            "is_exhausted": spend >= (self.monthly_budget * self.cutoff_threshold),
            "alert_triggered": spend >= (self.monthly_budget * self.alert_threshold),
        }
```

```python
    def get_routing_labels(self, limit: int = 500) -> list[dict]:
        """Get labeled routing decisions from feedback data for classifier training.

        Joins abby_messages (which stores routing metadata) with feedback
        to build a labeled dataset: message → route → feedback (good/bad).
        Returns up to `limit` labeled pairs.
        """
        try:
            with self.engine.connect() as conn:
                rows = conn.execute(
                    text("""
                        SELECT
                            m_user.content AS message,
                            m_asst.metadata->>'model' AS routed_model,
                            m_asst.metadata->>'reason' AS route_reason,
                            f.rating,
                            f.category
                        FROM app.abby_feedback f
                        JOIN app.abby_messages m_asst ON m_asst.id = f.message_id
                        JOIN app.abby_messages m_user ON m_user.conversation_id = m_asst.conversation_id
                            AND m_user.role = 'user'
                            AND m_user.created_at < m_asst.created_at
                        WHERE m_asst.metadata->>'model' IS NOT NULL
                        ORDER BY f.created_at DESC
                        LIMIT :limit
                    """),
                    {"limit": limit},
                ).fetchall()
                return [
                    {
                        "message": row[0],
                        "routed_model": row[1],
                        "route_reason": row[2],
                        "rating": row[3],
                        "category": row[4],
                    }
                    for row in rows
                ]
        except Exception:
            logger.exception("Failed to get routing labels")
            return []

    def get_routing_label_count(self) -> int:
        """Count available labeled routing decisions. Classifier can be trained at 500+."""
        try:
            with self.engine.connect() as conn:
                row = conn.execute(
                    text("""
                        SELECT COUNT(*)
                        FROM app.abby_feedback f
                        JOIN app.abby_messages m ON m.id = f.message_id
                        WHERE m.metadata->>'model' IS NOT NULL
                    """),
                ).fetchone()
                return int(row[0]) if row else 0
        except Exception:
            logger.exception("Failed to count routing labels")
            return 0
```

Note: If the `abby_feedback` table doesn't exist yet, this query will gracefully fail and return empty. The feedback endpoint already exists in the Laravel backend — this just reads from it.
