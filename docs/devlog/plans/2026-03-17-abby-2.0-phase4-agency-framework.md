# Abby 2.0 Phase 4: Agency Framework — Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Give Abby hands — she proposes multi-step action plans (create concept sets, build cohorts, generate patient counts), users confirm before execution, and every action is logged with rollback capability.

**Architecture:** Three subsystems: (1) Tool Registry — declarative tool definitions with risk levels, parameter schemas, and rollback strategies, (2) Plan-Confirm-Execute Engine — decomposes user requests into reviewable step-by-step plans, executes with checkpoints, (3) Action tools — first wave of 6 tools that call existing Laravel API endpoints via HTTP. The Python AI service orchestrates plans; the Laravel backend performs the actual data operations through its existing validated API.

**Tech Stack:** Python 3.12 (FastAPI, httpx), PHP 8.4 (Laravel 11), React 19 (TypeScript, Zustand), PostgreSQL 16

```python
"""Authenticated HTTP client for calling Laravel API endpoints.

Abby's agency tools execute actions by calling the existing Laravel API
rather than writing to the database directly. This ensures all validation,
authorization, and business logic is applied consistently.
"""
from __future__ import annotations

import logging
from typing import Any

import httpx

from app.config import settings

logger = logging.getLogger(__name__)


class AgencyApiClient:
    """HTTP client for Abby to call Laravel API endpoints on behalf of users."""

    def __init__(self, base_url: str | None = None) -> None:
        self.base_url = (base_url or settings.agency_api_base_url).rstrip("/")

    async def call(
        self,
        method: str,
        path: str,
        auth_token: str,
        data: dict | None = None,
        timeout: int = 30,
    ) -> dict[str, Any]:
        """Make an authenticated API call to the Laravel backend."""
        url = f"{self.base_url}/api/v1{path}"
        headers = {
            "Authorization": f"Bearer {auth_token}",
            "Accept": "application/json",
            "Content-Type": "application/json",
        }

        async with httpx.AsyncClient(timeout=timeout) as client:
            if method.upper() == "GET":
                response = await client.get(url, headers=headers)
            elif method.upper() == "POST":
                response = await client.post(url, headers=headers, json=data or {})
            elif method.upper() == "PUT":
                response = await client.put(url, headers=headers, json=data or {})
            elif method.upper() == "DELETE":
                response = await client.delete(url, headers=headers)
            else:
                raise ValueError(f"Unsupported HTTP method: {method}")

            if response.status_code >= 400:
                logger.error(
                    "API call failed: %s %s → %d: %s",
                    method, path, response.status_code, response.text[:500],
                )
                return {
                    "success": False,
                    "status": response.status_code,
                    "error": response.json().get("message", response.text[:200]),
                }

            return {
                "success": True,
                "status": response.status_code,
                "data": response.json(),
            }
```

```python
"""Action audit trail with checkpoint storage for rollback capability.

Every agency action (create, modify, delete) is logged with its parameters,
result, and checkpoint data. Checkpoint data stores the entity's prior state
so modifications can be rolled back.
"""
from __future__ import annotations

import json
import logging
from typing import Any

from sqlalchemy import text
from sqlalchemy.engine import Engine

logger = logging.getLogger(__name__)


class ActionLogger:
    """Logs agency actions to abby_action_log with checkpoint support."""

    def __init__(self, engine: Engine) -> None:
        self.engine = engine

    def log_action(
        self,
        user_id: int | None,
        action_type: str,
        tool_name: str,
        risk_level: str,
        parameters: dict | None = None,
        result: dict | None = None,
        plan: dict | None = None,
        checkpoint_data: dict | None = None,
    ) -> int | None:
        """Log an action and return the action log ID."""
        try:
            with self.engine.connect() as conn:
                row = conn.execute(
                    text("""
                        INSERT INTO app.abby_action_log
                            (user_id, action_type, tool_name, risk_level,
                             plan, parameters, result, checkpoint_data, created_at)
                        VALUES
                            (:user_id, :action_type, :tool_name, :risk_level,
                             :plan::jsonb, :parameters::jsonb, :result::jsonb,
                             :checkpoint_data::jsonb, NOW())
                        RETURNING id
                    """),
                    {
                        "user_id": user_id,
                        "action_type": action_type,
                        "tool_name": tool_name,
                        "risk_level": risk_level,
                        "plan": json.dumps(plan) if plan else None,
                        "parameters": json.dumps(parameters) if parameters else None,
                        "result": json.dumps(result) if result else None,
                        "checkpoint_data": json.dumps(checkpoint_data) if checkpoint_data else None,
                    },
                ).fetchone()
                conn.commit()
                return row[0] if row else None
        except Exception:
            logger.exception("Failed to log action")
            return None

    def get_recent_actions(self, user_id: int, limit: int = 20) -> list[dict]:
        """Get recent actions for a user."""
        try:
            with self.engine.connect() as conn:
                rows = conn.execute(
                    text("""
                        SELECT id, user_id, action_type, tool_name, risk_level,
                               plan, parameters, result, checkpoint_data, rolled_back, created_at
                        FROM app.abby_action_log
                        WHERE user_id = :user_id
                        ORDER BY created_at DESC
                        LIMIT :limit
                    """),
                    {"user_id": user_id, "limit": limit},
                ).fetchall()
                return [
                    {
                        "id": r[0], "user_id": r[1], "action_type": r[2],
                        "tool_name": r[3], "risk_level": r[4],
                        "plan": r[5], "parameters": r[6], "result": r[7],
                        "checkpoint_data": r[8], "rolled_back": r[9],
                        "created_at": str(r[10]),
                    }
                    for r in rows
                ]
        except Exception:
            logger.exception("Failed to get recent actions")
            return []

    def mark_rolled_back(self, action_id: int) -> None:
        """Mark an action as rolled back."""
        try:
            with self.engine.connect() as conn:
                conn.execute(
                    text("UPDATE app.abby_action_log SET rolled_back = TRUE WHERE id = :id"),
                    {"id": action_id},
                )
                conn.commit()
        except Exception:
            logger.exception("Failed to mark action as rolled back")
```

```python
"""Declarative tool registry — defines available actions with risk levels.

Each tool declares its name, description, risk level, confirmation requirements,
and rollback capability. The registry provides the tool catalog to the LLM
for plan generation and to the frontend for UI rendering.
"""
from __future__ import annotations

from dataclasses import dataclass, field
from enum import Enum
from typing import Any


class RiskLevel(Enum):
    LOW = "low"
    MEDIUM = "medium"
    HIGH = "high"


@dataclass
class ToolDefinition:
    name: str
    description: str
    risk_level: RiskLevel
    requires_confirmation: bool = True
    rollback_capable: bool = True
    parameters_schema: dict[str, Any] = field(default_factory=dict)


class ToolRegistry:
    """Registry of available action tools."""

    def __init__(self) -> None:
        self._tools: dict[str, ToolDefinition] = {}

    def register(self, tool: ToolDefinition) -> None:
        self._tools[tool.name] = tool

    def get(self, name: str) -> ToolDefinition | None:
        return self._tools.get(name)

    def list_tools(self) -> list[ToolDefinition]:
        return list(self._tools.values())

    def list_by_risk(self, risk_level: RiskLevel) -> list[ToolDefinition]:
        return [t for t in self._tools.values() if t.risk_level == risk_level]

    def format_for_prompt(self) -> str:
        """Format tool catalog for LLM prompt injection."""
        lines = ["Available actions:"]
        for tool in self._tools.values():
            confirm = "requires approval" if tool.requires_confirmation else "auto-execute"
            rollback = "rollback available" if tool.rollback_capable else "no rollback"
            lines.append(
                f"  - {tool.name} [{tool.risk_level.value.upper()}]: "
                f"{tool.description} ({confirm}, {rollback})"
            )
        return "\n".join(lines)

    @classmethod
    def default(cls) -> ToolRegistry:
        """Create registry with Phase 4 tools."""
        registry = cls()
        registry.register(ToolDefinition(
            name="create_concept_set",
            description="Create a new concept set with specified OMOP concepts",
            risk_level=RiskLevel.MEDIUM,
            parameters_schema={
                "name": "string (required)", "description": "string",
                "concepts": "list of {concept_id, include_descendants, is_excluded}",
            },
        ))
        registry.register(ToolDefinition(
            name="create_cohort_definition",
            description="Create a cohort definition with entry/inclusion/exclusion criteria",
            risk_level=RiskLevel.MEDIUM,
            parameters_schema={
                "name": "string (required)", "description": "string",
                "expression_json": "OHDSI CohortExpression JSON",
            },
        ))
        registry.register(ToolDefinition(
            name="generate_cohort",
            description="Execute cohort generation to compute patient count",
            risk_level=RiskLevel.MEDIUM,
            parameters_schema={
                "cohort_definition_id": "integer (required)",
                "source_id": "integer (required)",
            },
        ))
        registry.register(ToolDefinition(
            name="clone_cohort",
            description="Duplicate an existing cohort definition as a starting point",
            risk_level=RiskLevel.LOW,
            parameters_schema={"cohort_definition_id": "integer (required)"},
        ))
        registry.register(ToolDefinition(
            name="compare_cohorts",
            description="Side-by-side comparison of two cohort definitions",
            risk_level=RiskLevel.LOW,
            rollback_capable=False,
            requires_confirmation=False,
            parameters_schema={
                "cohort_id_a": "integer (required)",
                "cohort_id_b": "integer (required)",
            },
        ))
        registry.register(ToolDefinition(
            name="export_results",
            description="Export analysis or cohort results to downloadable format",
            risk_level=RiskLevel.LOW,
            rollback_capable=False,
            requires_confirmation=False,
            parameters_schema={
                "entity_type": "string (cohort|analysis)",
                "entity_id": "integer (required)",
            },
        ))
        return registry
```

```python
"""Plan-Confirm-Execute engine — orchestrates multi-step action plans.

Flow: User request → LLM generates plan → User reviews and approves →
Engine executes steps sequentially → Reports results.
"""
from __future__ import annotations

import time
import uuid
import logging
from dataclasses import dataclass, field
from enum import Enum
from typing import Any

from app.agency.tool_registry import ToolRegistry, RiskLevel
from app.config import settings

logger = logging.getLogger(__name__)


class PlanStatus(Enum):
    PENDING = "pending"
    APPROVED = "approved"
    EXECUTING = "executing"
    COMPLETED = "completed"
    FAILED = "failed"
    CANCELLED = "cancelled"


@dataclass
class PlanStep:
    tool_name: str
    parameters: dict[str, Any]
    status: str = "pending"  # pending, running, completed, failed, skipped
    result: dict[str, Any] | None = None
    error: str | None = None

    def to_dict(self) -> dict:
        return {
            "tool_name": self.tool_name,
            "parameters": self.parameters,
            "status": self.status,
            "result": self.result,
            "error": self.error,
        }


@dataclass
class ActionPlan:
    plan_id: str
    user_id: int
    description: str
    steps: list[PlanStep]
    status: PlanStatus = PlanStatus.PENDING
    created_at: float = field(default_factory=time.time)
    expires_at: float = 0.0
    auth_token: str = ""

    def to_dict(self) -> dict:
        return {
            "plan_id": self.plan_id,
            "user_id": self.user_id,
            "description": self.description,
            "steps": [s.to_dict() for s in self.steps],
            "status": self.status.value,
            "created_at": self.created_at,
            "expires_at": self.expires_at,
        }

    @property
    def is_expired(self) -> bool:
        return time.time() > self.expires_at


class PlanEngine:
    """Orchestrates Plan-Confirm-Execute flow for agency actions."""

    def __init__(
        self,
        tool_registry: ToolRegistry | None = None,
        action_logger: Any = None,
        api_client: Any = None,
        db_engine: Any = None,
    ) -> None:
        self.registry = tool_registry or ToolRegistry.default()
        self.action_logger = action_logger
        self.api_client = api_client
        self.db_engine = db_engine
        self._active_plans: dict[str, ActionPlan] = {}

    def create_plan(
        self,
        user_id: int,
        description: str,
        steps: list[PlanStep],
        auth_token: str = "",
    ) -> ActionPlan:
        """Create a new action plan for user review."""
        # Validate all tool names
        for step in steps:
            if self.registry.get(step.tool_name) is None:
                raise ValueError(f"Unknown tool: {step.tool_name}")

        plan = ActionPlan(
            plan_id=str(uuid.uuid4())[:8],
            user_id=user_id,
            description=description,
            steps=steps,
            expires_at=time.time() + settings.agency_plan_expiry_seconds,
            auth_token=auth_token,
        )

        self._active_plans[plan.plan_id] = plan
        return plan

    def approve_plan(self, plan: ActionPlan) -> None:
        """Mark a plan as approved by the user."""
        plan.status = PlanStatus.APPROVED

    def cancel_plan(self, plan: ActionPlan) -> None:
        """Cancel a plan."""
        plan.status = PlanStatus.CANCELLED

    def get_plan(self, plan_id: str) -> ActionPlan | None:
        """Retrieve an active plan by ID."""
        return self._active_plans.get(plan_id)

    async def execute_plan(self, plan: ActionPlan) -> ActionPlan:
        """Execute an approved plan step by step."""
        if plan.status != PlanStatus.APPROVED:
            raise ValueError(f"Plan must be approved before execution. Status: {plan.status.value}")

        if plan.is_expired:
            plan.status = PlanStatus.FAILED
            raise ValueError("Plan has expired. Please create a new plan.")

        plan.status = PlanStatus.EXECUTING

        for i, step in enumerate(plan.steps):
            step.status = "running"
            try:
                result = await self._execute_step(step, plan)
                step.result = result
                step.status = "completed"

                # Log the action
                if self.action_logger:
                    tool_def = self.registry.get(step.tool_name)
                    self.action_logger.log_action(
                        user_id=plan.user_id,
                        action_type="execute",
                        tool_name=step.tool_name,
                        risk_level=tool_def.risk_level.value if tool_def else "unknown",
                        plan=plan.to_dict(),
                        parameters=step.parameters,
                        result=result,
                    )

            except Exception as e:
                step.status = "failed"
                step.error = str(e)
                logger.exception("Plan step %d (%s) failed", i, step.tool_name)
                # Mark remaining steps as skipped
                for remaining in plan.steps[i + 1:]:
                    remaining.status = "skipped"
                plan.status = PlanStatus.FAILED
                return plan

        plan.status = PlanStatus.COMPLETED
        return plan

    async def _execute_step(self, step: PlanStep, plan: ActionPlan) -> dict:
        """Execute a single plan step by calling the appropriate tool."""
        if not self.api_client:
            raise RuntimeError("API client not configured")

        tool_name = step.tool_name
        params = step.parameters

        # Route to the appropriate tool implementation
        from app.agency.tools.concept_set_tools import execute_create_concept_set
        from app.agency.tools.cohort_tools import (
            execute_create_cohort_definition,
            execute_generate_cohort,
            execute_clone_cohort,
        )
        from app.agency.tools.query_tools import execute_compare_cohorts, execute_export_results

        tool_map = {
            "create_concept_set": execute_create_concept_set,
            "create_cohort_definition": execute_create_cohort_definition,
            "generate_cohort": execute_generate_cohort,
            "clone_cohort": execute_clone_cohort,
            "compare_cohorts": execute_compare_cohorts,
            "export_results": execute_export_results,
        }

        executor = tool_map.get(tool_name)
        if not executor:
            raise ValueError(f"No executor for tool: {tool_name}")

        return await executor(self.api_client, params, plan.auth_token)
```

```python
async def execute_create_concept_set(
    api_client: Any,
    params: dict,
    auth_token: str,
) -> dict:
    """Create a concept set and optionally add concept items."""
    name = params.get("name", "Untitled Concept Set")
    description = params.get("description", "")
    concepts = params.get("concepts", [])

    # Step 1: Create the concept set
    result = await api_client.call("POST", "/concept-sets", auth_token, data={
        "name": name,
        "description": description,
        "is_public": False,
    })

    if not result.get("success"):
        return {"success": False, "error": result.get("error", "Failed to create concept set")}

    concept_set_id = result["data"]["data"]["id"]

    # Step 2: Add concept items
    items_added = 0
    for concept in concepts:
        item_result = await api_client.call("POST", f"/concept-sets/{concept_set_id}/items", auth_token, data={
            "concept_id": concept.get("concept_id"),
            "include_descendants": concept.get("include_descendants", True),
            "is_excluded": concept.get("is_excluded", False),
            "include_mapped": concept.get("include_mapped", False),
        })
        if item_result.get("success"):
            items_added += 1

    return {
        "success": True,
        "concept_set_id": concept_set_id,
        "name": name,
        "items_added": items_added,
        "message": f"Created concept set '{name}' with {items_added} concepts",
    }
```

```python
    api_client: Any,
    params: dict,
    auth_token: str,

    """Queue cohort generation to compute patient count."""
    cohort_id = params.get("cohort_definition_id")
    source_id = params.get("source_id", 1)

    result = await api_client.call(
        "POST", f"/cohort-definitions/{cohort_id}/generate", auth_token,

        return {"success": False, "error": result.get("error", "Failed to queue generation")}

    gen_data = result["data"]["data"]
    return {
        "success": True,
        "generation_id": gen_data["id"],
        "status": gen_data.get("status", "queued"),
        "message": f"Cohort generation queued (ID: {gen_data['id']})",
    }

    api_client: Any,
    params: dict,
    auth_token: str,

    source_id = params.get("cohort_definition_id")

    source = await api_client.call("GET", f"/cohort-definitions/{source_id}", auth_token)

        "cohort_definition_id": result["data"]["data"]["id"],
```
