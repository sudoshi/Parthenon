# Abby 2.0 Phase 6: Institutional Intelligence — Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make the whole institution smarter — Abby automatically captures reusable knowledge from research activity, surfaces relevant discoveries to other researchers, and promotes frequently asked questions into an institutional knowledge base.

**Architecture:** Three components: (1) Knowledge capture pipeline — event listeners that extract artifacts (cohort patterns, analysis configs, data findings) from successful actions, (2) Knowledge surfacing — contextual suggestions based on similarity to current work, (3) FAQ auto-promotion — questions asked 3+ times across users become vetted institutional answers. All stored in `abby_knowledge_artifacts`, `abby_corrections`, and `abby_data_findings` tables with vector embeddings for semantic search.

**Tech Stack:** Python 3.12 (FastAPI, SQLAlchemy, sentence-transformers), PostgreSQL 16 (pgvector)

**Spec Reference:** `docs/superpowers/specs/2026-03-16-abby-2.0-cognitive-architecture-design.md` — Section 4 (Institutional Knowledge) + Phase 6 delivery section.

```python
"""Automatic knowledge artifact extraction from research activity.

Captures reusable patterns when researchers successfully create cohorts,
complete analyses, provide corrections, or discover data quality issues.
"""
from __future__ import annotations

import json
import logging
from dataclasses import dataclass
from typing import Any

from sqlalchemy import text
from sqlalchemy.engine import Engine

logger = logging.getLogger(__name__)


@dataclass
class KnowledgeArtifact:
    artifact_type: str
    title: str
    summary: str
    tags: list[str]
    disease_area: str
    artifact_data: dict


class KnowledgeCapture:
    """Captures and stores reusable knowledge artifacts."""

    def __init__(self, engine: Engine, embedder: Any = None) -> None:
        self.engine = engine
        self.embedder = embedder

    def _embed(self, text_content: str) -> str | None:
        if not self.embedder:
            return None
        try:
            vec = self.embedder.encode([text_content])[0]
            return "[" + ",".join(str(v) for v in vec) + "]"
        except Exception:
            return None

    def capture_cohort_creation(
        self, user_id: int, cohort_name: str, concept_ids: list[int],
        expression_summary: str, conversation_id: int | None = None,
    ) -> KnowledgeArtifact | None:
        """Capture a successful cohort creation as a reusable pattern."""
        artifact = KnowledgeArtifact(
            artifact_type="cohort_pattern", title=cohort_name,
            summary=expression_summary, tags=[], disease_area="",
            artifact_data={"concept_ids": concept_ids, "expression_summary": expression_summary},
        )
        embed_text = f"{cohort_name} {expression_summary}"
        self._store(artifact, user_id, conversation_id, embed_text)
        return artifact

    def capture_analysis_completion(
        self, user_id: int, analysis_type: str, cohort_name: str,
        config: dict, findings_summary: str,
    ) -> KnowledgeArtifact | None:
        """Capture a completed analysis configuration and findings."""
        title = f"{analysis_type.title()} of {cohort_name}"
        artifact = KnowledgeArtifact(
            artifact_type="analysis_config", title=title,
            summary=findings_summary, tags=[analysis_type], disease_area="",
            artifact_data={"config": config, "findings": findings_summary},
        )
        self._store(artifact, user_id, None, f"{title} {findings_summary}")
        return artifact

    def capture_correction(
        self, user_id: int, original: str, correction: str, context: dict | None = None,
    ) -> None:
        """Record a user correction for institutional learning."""
        try:
            with self.engine.connect() as conn:
                conn.execute(
                    text("""
                        INSERT INTO app.abby_corrections
                            (user_id, original_response, correction, context, created_at)
                        VALUES (:uid, :original, :correction, :context::jsonb, NOW())
                    """),
                    {"uid": user_id, "original": original, "correction": correction,
                     "context": json.dumps(context) if context else None},
                )
                conn.commit()
        except Exception:
            logger.exception("Failed to capture correction")

    def capture_data_finding(
        self, user_id: int, domain: str, tables: list[str],
        summary: str, severity: str = "info", workaround: str | None = None,
    ) -> None:
        """Record a data quality finding."""
        try:
            with self.engine.connect() as conn:
                conn.execute(
                    text("""
                        INSERT INTO app.abby_data_findings
                            (discovered_by, affected_domain, affected_tables, finding_summary,
                             severity, workaround, created_at)
                        VALUES (:uid, :domain, :tables, :summary, :severity, :workaround, NOW())
                    """),
                    {"uid": user_id, "domain": domain, "tables": tables,
                     "summary": summary, "severity": severity, "workaround": workaround},
                )
                conn.commit()
        except Exception:
            logger.exception("Failed to capture data finding")

    def search_similar(self, query: str, limit: int = 5) -> list[dict]:
        """Search knowledge artifacts by semantic similarity."""
        embedding = self._embed(query)
        if not embedding:
            return []
        try:
            with self.engine.connect() as conn:
                rows = conn.execute(
                    text("""
                        SELECT id, type, title, summary,
                               embedding <=> :embedding::vector AS distance,
                               usage_count
                        FROM app.abby_knowledge_artifacts
                        WHERE status = 'active' AND embedding IS NOT NULL
                        ORDER BY distance ASC LIMIT :limit
                    """),
                    {"embedding": embedding, "limit": limit},
                ).fetchall()
                return [{"id": r[0], "type": r[1], "title": r[2], "summary": r[3],
                         "distance": r[4], "usage_count": r[5]} for r in rows]
        except Exception:
            logger.exception("Failed to search artifacts")
            return []

    def increment_usage(self, artifact_id: int) -> None:
        """Increment usage count when an artifact is surfaced and accepted."""
        try:
            with self.engine.connect() as conn:
                conn.execute(
                    text("UPDATE app.abby_knowledge_artifacts SET usage_count = usage_count + 1 WHERE id = :id"),
                    {"id": artifact_id},
                )
                conn.commit()
        except Exception:
            logger.exception("Failed to increment usage")

    def _store(self, artifact: KnowledgeArtifact, user_id: int,
               conversation_id: int | None, embed_text: str) -> None:
        embedding = self._embed(embed_text)
        try:
            with self.engine.connect() as conn:
                conn.execute(
                    text("""
                        INSERT INTO app.abby_knowledge_artifacts
                            (type, title, summary, tags, disease_area, created_by,
                             source_conversation_id, artifact_data, embedding, created_at, updated_at)
                        VALUES (:type, :title, :summary, :tags, :disease_area, :uid,
                                :conv_id, :data::jsonb, :embedding::vector, NOW(), NOW())
                    """),
                    {"type": artifact.artifact_type, "title": artifact.title,
                     "summary": artifact.summary, "tags": artifact.tags,
                     "disease_area": artifact.disease_area, "uid": user_id,
                     "conv_id": conversation_id,
                     "data": json.dumps(artifact.artifact_data),
                     "embedding": embedding},
                )
                conn.commit()
        except Exception:
            logger.exception("Failed to store knowledge artifact")
```

```python
"""Contextual knowledge surfacing — suggests relevant institutional artifacts.

Proactively surfaces relevant knowledge when researchers are working on
topics where institutional artifacts exist. "3 other researchers built
similar cohorts — would you like to see their approach?"
"""
from __future__ import annotations

import logging
from typing import Any

from app.config import settings

logger = logging.getLogger(__name__)


class KnowledgeSurfacer:
    """Surfaces relevant institutional knowledge contextually."""

    def __init__(self, knowledge_capture: Any) -> None:
        self.knowledge_capture = knowledge_capture

    def suggest(
        self, query: str, max_results: int | None = None, max_distance: float = 0.5,
    ) -> list[dict]:
        """Find relevant knowledge artifacts for the current query."""
        max_results = max_results or settings.institutional_max_suggestions
        results = self.knowledge_capture.search_similar(query, limit=max_results * 2)
        filtered = [r for r in results if r.get("distance", 1.0) <= max_distance]
        return filtered[:max_results]

    def format_for_prompt(self, suggestions: list[dict]) -> str:
        """Format suggestions for system prompt injection."""
        if not suggestions:
            return ""
        lines = ["INSTITUTIONAL KNOWLEDGE (from other researchers):"]
        for s in suggestions:
            used = f", used {s['usage_count']}x" if s.get("usage_count", 0) > 0 else ""
            lines.append(f"  - [{s['type']}] {s['title']}: {s.get('summary', '')}{used}")
        lines.append("  Mention these to the user if relevant to their question.")
        return "\n".join(lines)
```

```python
"""FAQ auto-promoter — promotes frequently asked questions to institutional knowledge.

When the same question is asked 3+ times across users, it gets
promoted to a vetted institutional FAQ with the best answer.
"""
from __future__ import annotations

import logging
from typing import Any

from sqlalchemy import text
from sqlalchemy.engine import Engine

from app.config import settings

logger = logging.getLogger(__name__)


class FAQPromoter:
    """Detects and promotes frequently asked questions."""

    def __init__(
        self, engine: Engine, embedder: Any = None,
        threshold: int | None = None,
    ) -> None:
        self.engine = engine
        self.embedder = embedder
        self.threshold = threshold or settings.institutional_faq_threshold

    def check_and_promote(self, question: str, answer: str) -> bool:
        """Check if a question has been asked enough times to promote."""
        try:
            with self.engine.connect() as conn:
                # Count similar questions using text similarity
                row = conn.execute(
                    text("""
                        SELECT COUNT(DISTINCT c.user_id)
                        FROM app.abby_messages m
                        JOIN app.abby_conversations c ON c.id = m.conversation_id
                        WHERE m.role = 'user'
                          AND m.content ILIKE :pattern
                    """),
                    {"pattern": f"%{question[:50]}%"},
                ).fetchone()

                count = int(row[0]) if row else 0
                if count < self.threshold:
                    return False

                # Promote to knowledge artifact
                embedding = None
                if self.embedder:
                    try:
                        vec = self.embedder.encode([question])[0]
                        embedding = "[" + ",".join(str(v) for v in vec) + "]"
                    except Exception:
                        pass

                conn.execute(
                    text("""
                        INSERT INTO app.abby_knowledge_artifacts
                            (type, title, summary, artifact_data, embedding, status, created_at, updated_at)
                        VALUES ('faq', :title, :summary, :data::jsonb,
                                :embedding::vector, 'active', NOW(), NOW())
                        ON CONFLICT DO NOTHING
                    """),
                    {"title": question[:500], "summary": answer[:2000],
                     "data": '{"question": ' + repr(question[:500]) + ', "answer": ' + repr(answer[:2000]) + '}',
                     "embedding": embedding},
                )
                conn.commit()
                logger.info("Promoted FAQ: %s (asked by %d users)", question[:80], count)
                return True
        except Exception:
            logger.exception("FAQ promotion check failed")
            return False

    def get_faqs(self, limit: int = 20) -> list[dict]:
        """Get all promoted FAQs."""
        try:
            with self.engine.connect() as conn:
                rows = conn.execute(
                    text("""
                        SELECT id, type, title, summary, usage_count
                        FROM app.abby_knowledge_artifacts
                        WHERE type = 'faq' AND status = 'active'
                        ORDER BY usage_count DESC
                        LIMIT :limit
                    """),
                    {"limit": limit},
                ).fetchall()
                return [{"id": r[0], "type": r[1], "title": r[2],
                         "summary": r[3], "usage_count": r[4]} for r in rows]
        except Exception:
            logger.exception("Failed to get FAQs")
            return []
```
