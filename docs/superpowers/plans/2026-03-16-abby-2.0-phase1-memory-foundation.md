# Abby 2.0 Phase 1: Memory Foundation — Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Give Abby persistent memory — she remembers who each user is, what they're working on, and maintains coherent context across long conversations and sessions.

**Architecture:** Four components: (1) PostgreSQL-backed conversation archive replacing ChromaDB TTL storage, (2) user research profile auto-learning, (3) working memory with intent stack and scratch pad, (4) context assembly pipeline with ranked, budget-aware prompt construction for MedGemma. All changes are backward-compatible with Abby 1.0.

**Tech Stack:** Python 3.12 (FastAPI, SQLAlchemy, sentence-transformers), PHP 8.4 (Laravel 11, Eloquent), React 19 (TypeScript, Zustand, TanStack Query), PostgreSQL 16 (pgvector extension), ChromaDB (dual-read during migration)

**Spec Reference:** `docs/superpowers/specs/2026-03-16-abby-2.0-cognitive-architecture-design.md` — Section 1 (Four-Tier Memory System) + Phase 1 delivery section.

```python
"""Intent stack for working memory — tracks active topics across conversation turns."""
from __future__ import annotations

from dataclasses import dataclass, field
from typing import Any


@dataclass
class IntentEntry:
    topic: str
    first_turn: int
    last_active_turn: int


class IntentStack:
    """Bounded stack of active conversation topics with expiry."""

    def __init__(self, max_depth: int = 3, expiry_turns: int = 10) -> None:
        self.max_depth = max_depth
        self.expiry_turns = expiry_turns
        self.entries: list[IntentEntry] = []

    def push(self, topic: str, turn: int) -> None:
        """Add a new topic. Evicts oldest if at max depth."""
        # Check if topic already exists — refresh instead
        for entry in self.entries:
            if entry.topic == topic:
                entry.last_active_turn = turn
                return

        if len(self.entries) >= self.max_depth:
            self.entries.pop(0)

        self.entries.append(IntentEntry(
            topic=topic,
            first_turn=turn,
            last_active_turn=turn,
        ))

    def refresh(self, topic: str, turn: int) -> bool:
        """Update last_active_turn for an existing topic. Returns False if not found."""
        for entry in self.entries:
            if entry.topic == topic:
                entry.last_active_turn = turn
                return True
        return False

    def clear_and_set(self, topic: str, turn: int) -> None:
        """Explicit topic change — clear all and set new topic."""
        self.entries.clear()
        self.push(topic, turn)

    def prune(self, current_turn: int) -> None:
        """Remove entries that have expired based on turn inactivity."""
        self.entries = [
            e for e in self.entries
            if (current_turn - e.last_active_turn) <= self.expiry_turns
        ]

    def current_topic(self) -> str | None:
        """Return the most recent topic, or None if stack is empty."""
        if not self.entries:
            return None
        return self.entries[-1].topic

    def get_context_string(self) -> str:
        """Format active topics for prompt injection."""
        if not self.entries:
            return ""
        topics = [e.topic for e in self.entries]
        return "Active conversation topics: " + ", ".join(topics)

    def __len__(self) -> int:
        return len(self.entries)

    def to_dict(self) -> dict[str, Any]:
        """Serialize for session storage."""
        return {
            "max_depth": self.max_depth,
            "expiry_turns": self.expiry_turns,
            "entries": [
                {
                    "topic": e.topic,
                    "first_turn": e.first_turn,
                    "last_active_turn": e.last_active_turn,
                }
                for e in self.entries
            ],
        }

    @classmethod
    def from_dict(cls, data: dict[str, Any]) -> IntentStack:
        """Deserialize from session storage."""
        stack = cls(
            max_depth=data["max_depth"],
            expiry_turns=data["expiry_turns"],
        )
        stack.entries = [
            IntentEntry(
                topic=e["topic"],
                first_turn=e["first_turn"],
                last_active_turn=e["last_active_turn"],
            )
            for e in data["entries"]
        ]
        return stack
```

from typing import Any


@dataclass
class Artifact:
    key: str
    value: str
    version: int = 1


class ScratchPad:
    """Session-scoped storage for intermediate reasoning artifacts."""

    def __init__(self) -> None:
        self._artifacts: dict[str, Artifact] = {}

    def store(self, key: str, value: str) -> None:
        """Store or overwrite an artifact. Increments version on overwrite."""
        existing = self._artifacts.get(key)
        version = (existing.version + 1) if existing else 1
        self._artifacts[key] = Artifact(key=key, value=value, version=version)

    def get(self, key: str) -> str | None:
        """Retrieve an artifact by key."""
        artifact = self._artifacts.get(key)
        return artifact.value if artifact else None

    def get_version(self, key: str) -> int:
        """Get the current version of an artifact. Returns 0 if not found."""
        artifact = self._artifacts.get(key)
        return artifact.version if artifact else 0

    def list_keys(self) -> list[str]:
        """List all artifact keys."""
        return list(self._artifacts.keys())

    def clear(self) -> None:
        """Remove all artifacts."""
        self._artifacts.clear()

    def estimated_tokens(self) -> int:
        """Rough token estimate for all artifacts (~4 chars per token)."""
        total_chars = sum(
            len(a.key) + len(a.value) + 20  # 20 for formatting overhead
            for a in self._artifacts.values()
        )
        return total_chars // 4

    def get_context_string(self) -> str:
        """Format all artifacts for prompt injection."""
        if not self._artifacts:
            return ""
        parts = ["Working scratch pad:"]
        for artifact in self._artifacts.values():
            parts.append(f"[{artifact.key} v{artifact.version}]: {artifact.value}")
        return "\n".join(parts)

    def to_dict(self) -> dict[str, Any]:
        """Serialize for session storage."""
        return {
            "artifacts": {
                k: {"value": a.value, "version": a.version}
                for k, a in self._artifacts.items()
            }
        }

    @classmethod
    def from_dict(cls, data: dict[str, Any]) -> ScratchPad:
        """Deserialize from session storage."""
        pad = cls()
        for key, artifact_data in data.get("artifacts", {}).items():
            pad._artifacts[key] = Artifact(
                key=key,
                value=artifact_data["value"],
                version=artifact_data["version"],
            )
        return pad
```

"""Context assembly pipeline — ranked, budget-aware prompt construction for LLM calls."""
from __future__ import annotations

from dataclasses import dataclass, field
from enum import Enum
from typing import Any


class ContextTier(Enum):
    WORKING = "working"
    PAGE = "page"
    LIVE = "live"
    EPISODIC = "episodic"
    SEMANTIC = "semantic"
    INSTITUTIONAL = "institutional"


TIER_DISPLAY_NAMES = {
    ContextTier.WORKING: "Working Memory",
    ContextTier.PAGE: "Page Context",
    ContextTier.LIVE: "Live Database Context",
    ContextTier.EPISODIC: "User History",
    ContextTier.SEMANTIC: "Domain Knowledge",
    ContextTier.INSTITUTIONAL: "Institutional Knowledge",
}

# Tier ordering for prompt sections (highest priority first)
TIER_ORDER = [
    ContextTier.WORKING,
    ContextTier.PAGE,
    ContextTier.LIVE,
    ContextTier.EPISODIC,
    ContextTier.SEMANTIC,
    ContextTier.INSTITUTIONAL,
]

# Default per-tier budgets for MedGemma (4K effective context)
MEDGEMMA_TIER_BUDGETS = {
    ContextTier.WORKING: 1500,
    ContextTier.PAGE: 500,
    ContextTier.LIVE: 800,
    ContextTier.EPISODIC: 400,
    ContextTier.SEMANTIC: 600,
    ContextTier.INSTITUTIONAL: 200,
}

# Minimum tokens guaranteed for safety-critical content
SAFETY_MINIMUM_TOKENS = 200


@dataclass
class ContextPiece:
    tier: ContextTier
    content: str
    relevance: float  # 0.0 to 1.0
    tokens: int
    source: str = ""
    is_safety_critical: bool = False


class ContextAssembler:
    """Assembles context from multiple tiers into a budget-aware prompt."""

    def __init__(
        self,
        total_budget: int = 4000,
        tier_budgets: dict[ContextTier, int] | None = None,
    ) -> None:
        self.total_budget = total_budget
        self.tier_budgets = tier_budgets or {}

    @classmethod
    def for_medgemma(cls) -> ContextAssembler:
        """Factory for MedGemma-optimized context assembly."""
        return cls(total_budget=4000, tier_budgets=MEDGEMMA_TIER_BUDGETS)

    @classmethod
    def for_model(cls, model_name: str) -> ContextAssembler:
        """Factory for model-specific context assembly. Phase 2 will add Claude profile."""
        if model_name == "medgemma":
            return cls.for_medgemma()
        # Phase 2: add "claude" case with 128K budget and cloud-safety filtering
        raise ValueError(f"Unknown model profile: {model_name}. Available: medgemma")

    def assemble(self, pieces: list[ContextPiece]) -> list[ContextPiece]:
        """Select and rank context pieces within budget constraints."""
        if not pieces:
            return []

        # Step 1: Separate safety-critical pieces (guaranteed inclusion)
        safety_pieces = [p for p in pieces if p.is_safety_critical]
        regular_pieces = [p for p in pieces if not p.is_safety_critical]

        # Step 2: Sort regular pieces by relevance (descending)
        regular_pieces.sort(key=lambda p: p.relevance, reverse=True)

        # Step 3: Reserve budget for safety pieces
        safety_tokens = sum(p.tokens for p in safety_pieces)
        remaining_budget = self.total_budget - safety_tokens

        # Step 4: Fill remaining budget with highest-relevance pieces
        selected: list[ContextPiece] = []
        tier_usage: dict[ContextTier, int] = {}
        used_tokens = 0

        for piece in regular_pieces:
            # Check per-tier budget
            tier_limit = self.tier_budgets.get(piece.tier)
            current_tier_usage = tier_usage.get(piece.tier, 0)
            if tier_limit is not None and (current_tier_usage + piece.tokens) > tier_limit:
                continue

            # Check total budget
            if (used_tokens + piece.tokens) > remaining_budget:
                continue

            selected.append(piece)
            used_tokens += piece.tokens
            tier_usage[piece.tier] = current_tier_usage + piece.tokens

        # Step 5: Add safety pieces
        selected.extend(safety_pieces)

        # Step 6: Sort by tier order for consistent prompt structure
        tier_rank = {tier: i for i, tier in enumerate(TIER_ORDER)}
        selected.sort(key=lambda p: (tier_rank.get(p.tier, 99), -p.relevance))

        return selected

    def format_prompt(self, pieces: list[ContextPiece]) -> str:
        """Format assembled pieces into a structured prompt section."""
        if not pieces:
            return ""

        sections: dict[ContextTier, list[str]] = {}
        for piece in pieces:
            if piece.tier not in sections:
                sections[piece.tier] = []
            sections[piece.tier].append(piece.content)

        parts = []
        for tier in TIER_ORDER:
            if tier in sections:
                display_name = TIER_DISPLAY_NAMES[tier]
                parts.append(f"## {display_name}")
                parts.extend(sections[tier])
                parts.append("")

        return "\n".join(parts)
```

"""Profile learner — extracts user research profile from conversation patterns.

Uses keyword extraction and pattern matching (not LLM calls) to learn about
the user's research interests, expertise, and preferences over time.
"""
from __future__ import annotations

import re
from dataclasses import dataclass, field
from typing import Any


# Research domain keywords for interest detection
DOMAIN_KEYWORDS = {
    "diabetes": ["diabetes", "diabetic", "t1dm", "t2dm", "hba1c", "glucose", "insulin", "metformin"],
    "cardiovascular": ["cardiovascular", "cardiac", "heart", "hypertension", "stroke", "myocardial", "atrial"],
    "oncology": ["cancer", "oncology", "tumor", "neoplasm", "chemotherapy", "carcinoma", "melanoma"],
    "respiratory": ["respiratory", "pulmonary", "asthma", "copd", "pneumonia", "lung"],
    "neurology": ["neurological", "alzheimer", "parkinson", "epilepsy", "dementia", "seizure"],
    "mental_health": ["depression", "anxiety", "psychiatric", "bipolar", "schizophrenia", "mental health"],
    "infectious_disease": ["infection", "antibiotic", "viral", "bacterial", "sepsis", "covid", "hiv"],
    "rheumatology": ["arthritis", "rheumatoid", "lupus", "autoimmune", "joint"],
    "nephrology": ["kidney", "renal", "dialysis", "nephropathy", "creatinine"],
    "endocrinology": ["thyroid", "hormone", "endocrine", "pituitary", "adrenal"],
}

# Terse preference indicators
TERSE_INDICATORS = [
    r"just\s+(give|show|tell)\s+me",
    r"don'?t\s+need\s+(the\s+)?explanation",
    r"skip\s+(the\s+)?(detail|explanation)",
    r"be\s+(brief|concise|short)",
    r"tldr",
    r"cut\s+to\s+(the\s+)?chase",
]

# Correction indicators
CORRECTION_PATTERNS = [
    r"no,?\s+(I\s+)?(meant|mean)\s+",
    r"not\s+that,?\s+(I\s+)?(want|need)",
    r"actually,?\s+(I\s+)?(was|am)\s+",
    r"I\s+said\s+",
    r"that'?s\s+not\s+(what|right)",
    r"specifically\s+",
]

# Expertise level keywords (higher complexity = higher expertise)
EXPERT_KEYWORDS = [
    "propensity score", "immortal time bias", "hazard ratio", "kaplan-meier",
    "cox regression", "confounding", "causal inference", "phenotyping",
    "prevalent", "incident", "washout period", "index date", "time-at-risk",
    "negative control", "positive control", "calibration", "large-scale",
]

BEGINNER_KEYWORDS = [
    "what is a cohort", "how do I", "what does", "can you explain",
    "I don't understand", "what's the difference between",
    "help me understand", "I'm new to",
]


@dataclass
class UserProfile:
    research_interests: list[str] = field(default_factory=list)
    expertise_domains: dict[str, float] = field(default_factory=dict)
    interaction_preferences: dict[str, Any] = field(default_factory=dict)
    frequently_used: dict[str, list[str]] = field(default_factory=dict)
    interaction_count: int = 0

    def to_dict(self) -> dict[str, Any]:
        return {
            "research_interests": self.research_interests,
            "expertise_domains": self.expertise_domains,
            "interaction_preferences": self.interaction_preferences,
            "frequently_used": self.frequently_used,
            "interaction_count": self.interaction_count,
        }

    @classmethod
    def from_dict(cls, data: dict[str, Any]) -> UserProfile:
        return cls(
            research_interests=data.get("research_interests", []),
            expertise_domains=data.get("expertise_domains", {}),
            interaction_preferences=data.get("interaction_preferences", {}),
            frequently_used=data.get("frequently_used", {}),
            interaction_count=data.get("interaction_count", 0),
        )

    def get_context_string(self) -> str:
        """Format profile for prompt injection."""
        parts = []
        if self.research_interests:
            parts.append(f"Research interests: {', '.join(self.research_interests[:5])}")
        if self.expertise_domains:
            top_domains = sorted(self.expertise_domains.items(), key=lambda x: x[1], reverse=True)[:3]
            domain_str = ", ".join(f"{d} ({v:.0%})" for d, v in top_domains)
            parts.append(f"Expertise: {domain_str}")
        verbosity = self.interaction_preferences.get("verbosity", "normal")
        if verbosity != "normal":
            parts.append(f"Prefers {verbosity} responses")
        return "; ".join(parts) if parts else ""


class ProfileLearner:
    """Extracts user research profile from conversation patterns."""

    def __init__(
        self,
        min_interactions_for_calibration: int = 5,
        decay_factor: float = 0.85,
        max_interests: int = 10,
    ) -> None:
        self.min_interactions_for_calibration = min_interactions_for_calibration
        self.decay_factor = decay_factor
        self.max_interests = max_interests

    def learn_from_conversation(
        self,
        profile: UserProfile,
        messages: list[dict[str, str]],
    ) -> UserProfile:
        """Analyze conversation messages and return a NEW updated profile (immutable pattern)."""
        user_messages = [m["content"] for m in messages if m.get("role") == "user"]
        combined_text = " ".join(user_messages)  # Keep original case for entity extraction
        combined_lower = combined_text.lower()

        # Create a mutable copy to work with
        updated = UserProfile(
            research_interests=list(profile.research_interests),
            expertise_domains=dict(profile.expertise_domains),
            interaction_preferences=dict(profile.interaction_preferences),
            frequently_used={k: list(v) for k, v in profile.frequently_used.items()},
            interaction_count=profile.interaction_count + 1,
        )

        # Extract research interests (uses lowered text for keyword matching)
        self._learn_interests(updated, combined_lower)

        # Detect interaction preferences
        self._learn_preferences(updated, user_messages)

        # Track frequently used terms (uses original case for entity extraction)
        self._learn_frequently_used(updated, combined_text)

        # Calibrate expertise (only after enough interactions)
        if updated.interaction_count >= self.min_interactions_for_calibration:
            self._calibrate_expertise(updated, combined_lower)

        return updated

    def _learn_interests(self, profile: UserProfile, text: str) -> None:
        """Detect research domain interests from message text."""
        for domain, keywords in DOMAIN_KEYWORDS.items():
            if any(kw in text for kw in keywords):
                if domain not in profile.research_interests:
                    profile.research_interests.append(domain)

        # Trim to max, applying decay (keep recent, drop old)
        if len(profile.research_interests) > self.max_interests:
            profile.research_interests = profile.research_interests[-self.max_interests:]

    def _learn_preferences(self, profile: UserProfile, messages: list[str]) -> None:
        """Detect interaction style preferences."""
        for msg in messages:
            msg_lower = msg.lower()

            # Check for terse preference
            for pattern in TERSE_INDICATORS:
                if re.search(pattern, msg_lower):
                    profile.interaction_preferences["verbosity"] = "terse"
                    break

            # Check for corrections
            for pattern in CORRECTION_PATTERNS:
                match = re.search(pattern, msg_lower)
                if match:
                    corrections = profile.interaction_preferences.get("corrections", [])
                    corrections.append(msg[:200])  # Store truncated correction context
                    profile.interaction_preferences["corrections"] = corrections[-10:]  # Keep last 10
                    break

    def _learn_frequently_used(self, profile: UserProfile, text: str) -> None:
        """Track frequently mentioned clinical terms and entities."""
        terms = profile.frequently_used.get("terms", [])

        # Extract capitalized multi-word terms (likely entity names)
        entity_pattern = r'\b([A-Z][a-z]+(?:\s+[A-Za-z]+){1,4})\b'
        # Look for quoted terms
        quoted_pattern = r'["\']([^"\']+)["\']'

        for pattern in [entity_pattern, quoted_pattern]:
            matches = re.findall(pattern, text)
            for match in matches:
                match_lower = match.lower().strip()
                if len(match_lower) > 3 and match_lower not in terms:
                    terms.append(match_lower)

        profile.frequently_used["terms"] = terms[-20:]  # Keep last 20

    def _calibrate_expertise(self, profile: UserProfile, text: str) -> None:
        """Calibrate expertise level based on vocabulary complexity."""
        expert_score = sum(1 for kw in EXPERT_KEYWORDS if kw in text)
        beginner_score = sum(1 for kw in BEGINNER_KEYWORDS if kw in text)

        if expert_score > 0 or beginner_score > 0:
            # Normalize to 0-1 range
            total = expert_score + beginner_score
            level = expert_score / total if total > 0 else 0.5

            # Apply decay weighting with existing score
            current = profile.expertise_domains.get("general", 0.5)
            updated = (self.decay_factor * current) + ((1 - self.decay_factor) * level)
            profile.expertise_domains["general"] = round(updated, 2)

```python
"""PostgreSQL-backed conversation store with pgvector similarity search.

Replaces ChromaDB per-user collections with PostgreSQL + pgvector for permanent,
queryable conversation storage. During migration, used alongside ChromaDB via
the migration_bridge module.
"""
from __future__ import annotations

import logging
from dataclasses import dataclass
from typing import Any

from sqlalchemy import text
from sqlalchemy.engine import Engine

logger = logging.getLogger(__name__)


@dataclass
class ConversationResult:
    message_id: int
    role: str
    content: str
    distance: float
    conversation_id: int


class ConversationStore:
    """PostgreSQL-backed conversation storage with vector similarity search."""

    def __init__(self, engine: Engine, embedder: Any) -> None:
        self.engine = engine
        self.embedder = embedder

    def store_message(
        self,
        conversation_id: int,
        role: str,
        content: str,
    ) -> None:
        """Store a message with its embedding vector."""
        try:
            embedding = self.embedder.encode([content])[0]
            embedding_str = "[" + ",".join(str(v) for v in embedding) + "]"

            with self.engine.connect() as conn:
                conn.execute(
                    text("""
                        UPDATE app.abby_messages
                        SET embedding = :embedding::vector
                        WHERE ctid = (
                            SELECT ctid FROM app.abby_messages
                            WHERE conversation_id = :conversation_id
                              AND role = :role
                              AND content = :content
                              AND embedding IS NULL
                            LIMIT 1
                        )
                    """),
                    {
                        "embedding": embedding_str,
                        "conversation_id": conversation_id,
                        "role": role,
                        "content": content,
                    },
                )
                conn.commit()
        except Exception:
            logger.exception("Failed to store message embedding")

    def search_similar(
        self,
        query: str,
        user_id: int,
        limit: int = 5,
        max_distance: float = 0.5,
    ) -> list[ConversationResult]:
        """Find similar past messages using cosine distance."""
        try:
            query_embedding = self.embedder.encode([query])[0]
            embedding_str = "[" + ",".join(str(v) for v in query_embedding) + "]"

            with self.engine.connect() as conn:
                rows = conn.execute(
                    text("""
                        SELECT m.id, m.role, m.content,
                               m.embedding <=> :embedding::vector AS distance,
                               m.conversation_id
                        FROM app.abby_messages m
                        JOIN app.abby_conversations c ON c.id = m.conversation_id
                        WHERE c.user_id = :user_id
                          AND m.embedding IS NOT NULL
                          AND (m.embedding <=> :embedding::vector) <= :max_distance
                        ORDER BY distance ASC
                        LIMIT :limit
                    """),
                    {
                        "embedding": embedding_str,
                        "user_id": user_id,
                        "limit": limit,
                        "max_distance": max_distance,
                    },
                ).fetchall()

            return [
                ConversationResult(
                    message_id=row[0],
                    role=row[1],
                    content=row[2],
                    distance=row[3],
                    conversation_id=row[4],
                )
                for row in rows
            ]
        except Exception:
            logger.exception("Failed to search similar messages")
            return []

    def get_recent(
        self,
        user_id: int,
        limit: int = 10,
    ) -> list[dict[str, Any]]:
        """Get recent messages for a user, ordered by time."""
        try:
            with self.engine.connect() as conn:
                rows = conn.execute(
                    text("""
                        SELECT m.id, m.role, m.content, m.created_at
                        FROM app.abby_messages m
                        JOIN app.abby_conversations c ON c.id = m.conversation_id
                        WHERE c.user_id = :user_id
                        ORDER BY m.created_at DESC
                        LIMIT :limit
                    """),
                    {"user_id": user_id, "limit": limit},
                ).fetchall()

            return [
                {
                    "id": row[0],
                    "role": row[1],
                    "content": row[2],
                    "created_at": str(row[3]),
                }
                for row in rows
            ]
        except Exception:
            logger.exception("Failed to get recent messages")
            return []
```

```python
"""Conversation summarizer — compresses older turns to manage context window capacity.

Uses MedGemma for summarization (zero additional cost, fast). Summaries are tagged
as synthetic so users can expand to see original turns if needed.
"""
from __future__ import annotations

import logging
from typing import Any

logger = logging.getLogger(__name__)


class ConversationSummarizer:
    """Manages conversation context window by summarizing older turns."""

    def __init__(
        self,
        threshold_ratio: float = 0.7,
        context_window: int = 4000,
    ) -> None:
        self.threshold_ratio = threshold_ratio
        self.context_window = context_window
        self.threshold_tokens = int(context_window * threshold_ratio)

    def estimate_tokens(self, messages: list[dict[str, Any]]) -> int:
        """Rough token estimate for a list of messages (~4 chars per token)."""
        total_chars = sum(len(m.get("content", "")) for m in messages)
        return total_chars // 4

    def should_summarize(self, messages: list[dict[str, Any]]) -> bool:
        """Check if conversation has exceeded the summarization threshold."""
        return self.estimate_tokens(messages) > self.threshold_tokens

    def split_for_summarization(
        self,
        messages: list[dict[str, Any]],
        keep_recent: int = 4,
    ) -> tuple[list[dict[str, Any]], list[dict[str, Any]]]:
        """Split messages into old (to summarize) and recent (to keep verbatim).

        keep_recent is the number of complete turns (user+assistant pairs) to keep.
        """
        # Keep the last `keep_recent * 2` messages (each turn = user + assistant)
        keep_count = keep_recent * 2
        if len(messages) <= keep_count:
            return [], messages

        split_point = len(messages) - keep_count
        return messages[:split_point], messages[split_point:]

    def format_summary_prompt(self, messages: list[dict[str, Any]]) -> str:
        """Create a prompt for MedGemma to summarize conversation history."""
        conversation = []
        for msg in messages:
            role = msg.get("role", "unknown")
            content = msg.get("content", "")
            conversation.append(f"{role}: {content}")

        joined = "\n".join(conversation)

        return (
            "Summarize this conversation history concisely. "
            "Preserve: key topics discussed, decisions made, important facts shared by the user, "
            "and any corrections or preferences expressed. "
            "Do not include greetings or filler. Be factual and brief.\n\n"
            f"Conversation:\n{joined}\n\n"
            "Summary:"
        )

    def create_summary_message(self, summary_text: str, original_count: int) -> dict[str, Any]:
        """Create a synthetic summary message to replace older turns."""
        return {
            "role": "system",
            "content": f"[Summary of {original_count} earlier messages]\n{summary_text}",
            "is_summary": True,
            "original_message_count": original_count,
        }
```

```python
"""Migration bridge — dual-read from PostgreSQL (primary) and ChromaDB (fallback).

During the ChromaDB → PostgreSQL migration period (2 weeks), this bridge queries
PostgreSQL first and falls back to ChromaDB for data not yet migrated. After
cutover, dual_read is set to False and ChromaDB is no longer queried.
"""
from __future__ import annotations

import logging
from dataclasses import dataclass
from typing import Any, Callable

logger = logging.getLogger(__name__)


@dataclass
class SearchResult:
    content: str
    distance: float
    role: str
    source: str  # "postgresql" or "chromadb"


class MigrationBridge:
    """Dual-read bridge for ChromaDB → PostgreSQL conversation migration."""

    def __init__(
        self,
        pg_store: Any,  # ConversationStore
        chroma_query_fn: Callable | None = None,
        dual_read: bool = True,
    ) -> None:
        self.pg_store = pg_store
        self.chroma_query_fn = chroma_query_fn
        self.dual_read = dual_read

    def search(
        self,
        query: str,
        user_id: int,
        limit: int = 5,
    ) -> list[SearchResult]:
        """Search for similar conversations, preferring PostgreSQL results."""
        results: list[SearchResult] = []

        # Step 1: Query PostgreSQL (primary)
        try:
            pg_results = self.pg_store.search_similar(
                query=query,
                user_id=user_id,
                limit=limit,
            )
            for r in pg_results:
                results.append(SearchResult(
                    content=r.content,
                    distance=r.distance,
                    role=r.role,
                    source="postgresql",
                ))
        except Exception:
            logger.exception("PostgreSQL conversation search failed")

        # Step 2: If dual_read enabled and PG returned fewer than limit, try ChromaDB
        if self.dual_read and len(results) < limit and self.chroma_query_fn:
            try:
                remaining = limit - len(results)
                chroma_results = self.chroma_query_fn(
                    query=query,
                    user_id=user_id,
                    limit=remaining,
                )
                for r in chroma_results:
                    content = r.get("content", "") if isinstance(r, dict) else r.content
                    distance = r.get("distance", 0.5) if isinstance(r, dict) else r.distance
                    results.append(SearchResult(
                        content=content,
                        distance=distance,
                        role="unknown",
                        source="chromadb",
                    ))
            except Exception:
                logger.exception("ChromaDB fallback search failed")

        # Step 3: Deduplicate by content similarity (exact match for now)
        seen_content: set[str] = set()
        deduped: list[SearchResult] = []
        for r in results:
            # Normalize for comparison
            normalized = r.content.strip().lower()[:200]
            if normalized not in seen_content:
                seen_content.add(normalized)
                deduped.append(r)

        # Step 4: Sort by distance (best matches first) and limit
        deduped.sort(key=lambda r: r.distance)
        return deduped[:limit]
```

import {
  fetchAbbyProfile,
  updateAbbyProfile,
  resetAbbyProfile,
} from '../../commons/services/abbyService';
import type { AbbyProfileUpdateRequest } from '../types/memory';

export function useAbbyProfile() {
  const queryClient = useQueryClient();

  const profileQuery = useQuery({
    queryKey: ['abby-profile'],
    queryFn: fetchAbbyProfile,
    staleTime: 5 * 60 * 1000, // 5 minutes
  });

  const updateMutation = useMutation({
    mutationFn: (data: AbbyProfileUpdateRequest) => updateAbbyProfile(data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['abby-profile'] });
    },
  });

  const resetMutation = useMutation({
    mutationFn: resetAbbyProfile,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['abby-profile'] });
    },
  });

  return {
    profile: profileQuery.data?.data,
    isLoading: profileQuery.isLoading,
    error: profileQuery.error,
    updateProfile: updateMutation.mutate,
    resetProfile: resetMutation.mutate,
    isUpdating: updateMutation.isPending,
  };
}

**Files:**

```python
"""One-time migration: export ChromaDB conversation collections to PostgreSQL.

Usage: python -m ai.scripts.migrate_chroma_to_pg [--dry-run] [--user-id N]

Reads from ChromaDB per-user collections (conversations_user_{id}) and writes
embeddings to the abby_messages.embedding column in PostgreSQL.
"""
import argparse
import logging
import sys

from sqlalchemy import create_engine, text

from ai.app.config import settings
from ai.app.chroma.client import get_chroma_client

logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")
logger = logging.getLogger(__name__)


def migrate_user_collection(
    chroma_client,
    engine,
    user_id: int,
    dry_run: bool = False,
) -> dict:
    """Migrate a single user's ChromaDB collection to PostgreSQL embeddings."""
    collection_name = f"conversations_user_{user_id}"
    stats = {"user_id": user_id, "chroma_count": 0, "matched": 0, "updated": 0, "errors": 0}

    try:
        collection = chroma_client.get_collection(collection_name)
    except Exception:
        logger.info(f"No collection found for user {user_id}, skipping")
        return stats

    # Get all documents with embeddings
    results = collection.get(include=["documents", "embeddings", "metadatas"])
    stats["chroma_count"] = len(results["ids"])

    if dry_run:
        logger.info(f"[DRY RUN] User {user_id}: {stats['chroma_count']} documents would be migrated")
        return stats

    with engine.connect() as conn:
        for i, doc_id in enumerate(results["ids"]):
            try:
                content = results["documents"][i]
                embedding = results["embeddings"][i]

                if not embedding or not content:
                    continue

                embedding_str = "[" + ",".join(str(v) for v in embedding) + "]"

                # Match by content and user, update embedding
                result = conn.execute(
                    text("""
                        UPDATE app.abby_messages m
                        SET embedding = :embedding::vector,
                            embedding_model = 'all-MiniLM-L6-v2'
                        FROM app.abby_conversations c
                        WHERE m.conversation_id = c.id
                          AND c.user_id = :user_id
                          AND m.content = :content
                          AND m.embedding IS NULL
                    """),
                    {
                        "embedding": embedding_str,
                        "user_id": user_id,
                        "content": content[:10000],  # Truncate for safety
                    },
                )
                if result.rowcount > 0:
                    stats["updated"] += result.rowcount
                stats["matched"] += 1

            except Exception as e:
                stats["errors"] += 1
                logger.warning(f"Error migrating doc {doc_id} for user {user_id}: {e}")

        conn.commit()

    logger.info(
        f"User {user_id}: {stats['chroma_count']} in ChromaDB, "
        f"{stats['matched']} matched, {stats['updated']} updated, "
        f"{stats['errors']} errors"
    )
    return stats


def main():
    parser = argparse.ArgumentParser(description="Migrate ChromaDB conversations to PostgreSQL")
    parser.add_argument("--dry-run", action="store_true", help="Show what would be migrated without writing")
    parser.add_argument("--user-id", type=int, help="Migrate a specific user only")
    args = parser.parse_args()

    engine = create_engine(settings.database_url)
    chroma_client = get_chroma_client()

    # Get all user collections
    if args.user_id:
        user_ids = [args.user_id]
    else:
        collections = chroma_client.list_collections()
        user_ids = []
        for col in collections:
            name = col.name if hasattr(col, "name") else str(col)
            if name.startswith("conversations_user_"):
                try:
                    uid = int(name.replace("conversations_user_", ""))
                    user_ids.append(uid)
                except ValueError:
                    continue

    logger.info(f"Found {len(user_ids)} user collections to migrate")

    total_stats = {"total_chroma": 0, "total_updated": 0, "total_errors": 0}
    for user_id in sorted(user_ids):
        stats = migrate_user_collection(chroma_client, engine, user_id, dry_run=args.dry_run)
        total_stats["total_chroma"] += stats["chroma_count"]
        total_stats["total_updated"] += stats["updated"]
        total_stats["total_errors"] += stats["errors"]

    logger.info(f"\n=== Migration Summary ===")
    logger.info(f"Users processed: {len(user_ids)}")
    logger.info(f"ChromaDB documents: {total_stats['total_chroma']}")
    logger.info(f"PostgreSQL updated: {total_stats['total_updated']}")
    logger.info(f"Errors: {total_stats['total_errors']}")

    if total_stats["total_errors"] > 0:
        sys.exit(1)


if __name__ == "__main__":
    main()
```
