```python
"""DAG-based parallel step execution with dependency tracking.

Decomposes multi-step plans into waves of independent steps that can
run concurrently via asyncio.gather. Steps with dependencies wait
until their prerequisites complete.
"""
from __future__ import annotations

import asyncio
import logging
from dataclasses import dataclass, field
from typing import Any, Callable, Awaitable

logger = logging.getLogger(__name__)


@dataclass
class DAGStep:
    id: str
    tool_name: str
    parameters: dict[str, Any]
    depends_on: list[str] = field(default_factory=list)
    status: str = "pending"
    result: dict | None = None

    def to_dict(self) -> dict:
        return {
            "id": self.id, "tool_name": self.tool_name,
            "parameters": self.parameters, "depends_on": self.depends_on,
            "status": self.status, "result": self.result,
        }


@dataclass
class DAGPlan:
    steps: list[DAGStep]

    def get_execution_waves(self) -> list[list[str]]:
        """Topological sort into waves of independent steps."""
        # Build adjacency and in-degree maps
        in_degree: dict[str, int] = {s.id: 0 for s in self.steps}
        dependents: dict[str, list[str]] = {s.id: [] for s in self.steps}
        step_ids = {s.id for s in self.steps}

        for step in self.steps:
            for dep in step.depends_on:
                if dep not in step_ids:
                    raise ValueError(f"Unknown dependency: {dep}")
                in_degree[step.id] += 1
                dependents[dep].append(step.id)

        waves: list[list[str]] = []
        remaining = dict(in_degree)

        while remaining:
            # Find all steps with zero in-degree
            wave = [sid for sid, deg in remaining.items() if deg == 0]
            if not wave:
                raise ValueError("Dependency cycle detected in plan")
            waves.append(sorted(wave))

            # Remove completed steps and update in-degrees
            for sid in wave:
                del remaining[sid]
                for dep_id in dependents.get(sid, []):
                    if dep_id in remaining:
                        remaining[dep_id] -= 1

        return waves

    def to_dict(self) -> dict:
        return {"steps": [s.to_dict() for s in self.steps]}


StepExecutor = Callable[[str, str, dict], Awaitable[dict]]


class DAGExecutor:
    """Executes DAG plans with parallel wave execution."""

    async def execute(
        self,
        plan: DAGPlan,
        step_executor: StepExecutor,
    ) -> dict[str, dict]:
        """Execute plan steps in dependency-ordered waves."""
        results: dict[str, dict] = {}
        failed_steps: set[str] = set()

        try:
            waves = plan.get_execution_waves()
        except ValueError as e:
            logger.error("Invalid DAG plan: %s", e)
            raise

        for wave in waves:
            # Filter out steps whose dependencies failed
            executable = []
            for step_id in wave:
                step = next(s for s in plan.steps if s.id == step_id)
                if any(dep in failed_steps for dep in step.depends_on):
                    results[step_id] = {"skipped": True, "reason": "dependency_failed"}
                    step.status = "skipped"
                else:
                    executable.append(step)

            if not executable:
                continue

            # Execute all steps in this wave concurrently
            async def run_step(s: DAGStep) -> tuple[str, dict]:
                try:
                    s.status = "running"
                    result = await step_executor(s.id, s.tool_name, s.parameters)
                    s.status = "completed"
                    s.result = result
                    return s.id, result
                except Exception as e:
                    s.status = "failed"
                    failed_steps.add(s.id)
                    return s.id, {"success": False, "error": str(e)}

            wave_results = await asyncio.gather(
                *(run_step(s) for s in executable),
                return_exceptions=False,
            )

            for step_id, result in wave_results:
                results[step_id] = result
                if not result.get("success", True):
                    failed_steps.add(step_id)

        return results
```

```python
"""Dry run mode — simulate actions without executing them.

For high-risk actions, dry run reports what WOULD happen without making
any changes. Uses descriptions of the tool's effect based on parameters.
"""
from __future__ import annotations

from typing import Any, TYPE_CHECKING

if TYPE_CHECKING:
    from app.agency.plan_engine import PlanStep


TOOL_DESCRIPTIONS = {
    "create_concept_set": lambda p: {
        "simulated": True, "would_create": "concept_set",
        "name": p.get("name", "Untitled"),
        "concept_count": len(p.get("concepts", [])),
        "description": f"Would create concept set '{p.get('name', 'Untitled')}' with {len(p.get('concepts', []))} concepts",
    },
    "create_cohort_definition": lambda p: {
        "simulated": True, "would_create": "cohort_definition",
        "name": p.get("name", "Untitled"),
        "description": f"Would create cohort definition '{p.get('name', 'Untitled')}'",

        "simulated": True, "would_execute": "cohort_generation",
        "cohort_definition_id": p.get("cohort_definition_id"),
        "source_id": p.get("source_id"),
        "description": f"Would generate cohort {p.get('cohort_definition_id')} against source {p.get('source_id')}",
    },
    "clone_cohort": lambda p: {
        "simulated": True, "would_create": "cohort_definition_clone",
        "source_id": p.get("cohort_definition_id"),
        "description": f"Would clone cohort {p.get('cohort_definition_id')}",
    },
    "modify_concept_set": lambda p: {
        "simulated": True, "would_modify": "concept_set",
        "concept_set_id": p.get("concept_set_id"),
        "description": f"Would modify concept set {p.get('concept_set_id')}",
    },
    "modify_cohort_criteria": lambda p: {
        "simulated": True, "would_modify": "cohort_definition",
        "cohort_definition_id": p.get("cohort_definition_id"),
        "description": f"Would modify cohort {p.get('cohort_definition_id')} criteria",
    },
    "execute_sql": lambda p: {
        "simulated": True, "would_execute": "sql_query", "read_only": True,
        "sql_preview": (p.get("sql", "")[:200] + "..." if len(p.get("sql", "")) > 200 else p.get("sql", "")),
        "description": f"Would execute read-only SQL query",
    },
    "run_characterization": lambda p: {
        "simulated": True, "would_execute": "characterization_analysis",
        "description": f"Would queue characterization analysis",
    },
    "run_incidence_analysis": lambda p: {
        "simulated": True, "would_execute": "incidence_rate_analysis",
        "description": f"Would queue incidence rate analysis",
    },
    "compare_cohorts": lambda p: {
        "simulated": True, "would_execute": "comparison", "read_only": True,
        "description": f"Would compare cohorts {p.get('cohort_id_a')} and {p.get('cohort_id_b')}",
    },
    "export_results": lambda p: {
        "simulated": True, "would_execute": "export", "read_only": True,
        "description": f"Would export {p.get('entity_type')} {p.get('entity_id')}",
    },
}


class DryRunSimulator:
    """Simulates plan execution without making changes."""

    def simulate(self, step: "PlanStep") -> dict[str, Any]:
        """Simulate a single step."""
        descriptor = TOOL_DESCRIPTIONS.get(step.tool_name)
        if descriptor:
            return descriptor(step.parameters)
        return {
            "simulated": True,
            "note": f"Unknown tool '{step.tool_name}' — cannot predict outcome",
        }

    def simulate_plan(self, steps: list["PlanStep"]) -> list[dict[str, Any]]:
        """Simulate all steps in a plan."""
        return [self.simulate(step) for step in steps]
```

```python
from __future__ import annotations
import logging
from typing import Any

logger = logging.getLogger(__name__)


async def execute_modify_concept_set(api_client: Any, params: dict, auth_token: str) -> dict:
    """Add or remove concepts from an existing concept set."""
    cs_id = params.get("concept_set_id")
    add_concepts = params.get("add_concepts", [])
    remove_ids = params.get("remove_concept_ids", [])

    added = 0
    for concept in add_concepts:
        result = await api_client.call("POST", f"/concept-sets/{cs_id}/items", auth_token, data={
            "concept_id": concept.get("concept_id"),
            "include_descendants": concept.get("include_descendants", True),
            "is_excluded": concept.get("is_excluded", False),
        })
        if result.get("success"):
            added += 1

    removed = 0
    for item_id in remove_ids:
        result = await api_client.call("DELETE", f"/concept-sets/{cs_id}/items/{item_id}", auth_token)
        if result.get("success"):
            removed += 1

    return {"success": True, "added": added, "removed": removed,
            "message": f"Modified concept set {cs_id}: +{added} -{removed} concepts"}


async def execute_modify_cohort_criteria(api_client: Any, params: dict, auth_token: str) -> dict:
    """Update the expression JSON for a cohort definition."""
    cohort_id = params.get("cohort_definition_id")
    result = await api_client.call("PUT", f"/cohort-definitions/{cohort_id}", auth_token, data={
        "expression_json": params.get("expression_json", {}),
    })
    if not result.get("success"):
        return {"success": False, "error": result.get("error", "Failed to update cohort")}
    return {"success": True, "cohort_definition_id": cohort_id,
            "message": f"Updated cohort {cohort_id} criteria"}
```

```python
"""SQL execution tool — validated read-only queries against CDM."""
from __future__ import annotations
import re
import logging
from typing import Any

logger = logging.getLogger(__name__)

DANGEROUS_PATTERNS = [
    re.compile(r'\b(INSERT|UPDATE|DELETE|DROP|ALTER|CREATE|TRUNCATE|GRANT|REVOKE)\b', re.IGNORECASE),
    re.compile(r'\b(pg_|information_schema|COPY|\\copy)\b', re.IGNORECASE),
]


def validate_sql_safety(sql: str) -> bool:
    """Validate SQL is read-only and safe to execute."""
    for pattern in DANGEROUS_PATTERNS:
        if pattern.search(sql):
            return False
    return True


async def execute_sql(api_client: Any, params: dict, auth_token: str) -> dict:
    """Execute a validated read-only SQL query."""
    sql = params.get("sql", "")
    if not validate_sql_safety(sql):
        return {"success": False, "error": "SQL blocked: contains prohibited DML/DDL statements"}

    result = await api_client.call("POST", "/abby/execute-sql", auth_token, data={"sql": sql}, timeout=60)
    if not result.get("success"):
        return {"success": False, "error": result.get("error", "SQL execution failed")}
    return {"success": True, "data": result.get("data", {}),
            "message": "SQL query executed successfully"}
```

```python
"""Pre-built workflow templates for common study designs.

Templates generate lists of PlanStep-compatible dicts that can be
passed to PlanEngine.create_plan(). Each template encodes OHDSI
best practices for study design.
"""
from __future__ import annotations
from typing import Any


class WorkflowTemplates:
    """Factory for common OHDSI study design workflows."""

    @staticmethod
    def incident_cohort(
        condition_name: str,
        condition_concepts: list[int],
        drug_name: str | None = None,
        drug_concepts: list[int] | None = None,
        washout_days: int = 365,
        source_id: int = 1,
    ) -> list[dict[str, Any]]:
        """New user / incident cohort design."""
        steps = [
            {
                "tool_name": "create_concept_set",
                "parameters": {
                    "name": f"{condition_name} Conditions",
                    "concepts": [{"concept_id": c, "include_descendants": True} for c in condition_concepts],
                },
            },
        ]
        if drug_name and drug_concepts:
            steps.append({
                "tool_name": "create_concept_set",
                "parameters": {
                    "name": f"{drug_name} Exposures",
                    "concepts": [{"concept_id": c, "include_descendants": True} for c in drug_concepts],
                },
            })
        steps.extend([
            {
                "tool_name": "create_cohort_definition",
                "parameters": {
                    "name": f"Incident {condition_name}" + (f" on {drug_name}" if drug_name else ""),
                    "description": f"New users with {condition_name}" + (f" and {drug_name} exposure" if drug_name else ""),
                    "expression_json": {
                        "PrimaryCriteria": {
                            "CriteriaList": [{"ConditionOccurrence": {"CodesetId": 0, "First": True}}],
                            "ObservationWindow": {"PriorDays": washout_days, "PostDays": 0},
                        },
                        "QualifiedLimit": {"Type": "First"},
                        "ExpressionLimit": {"Type": "First"},
                    },
                },
            },
            {
                "tool_name": "generate_cohort",
                "parameters": {"source_id": source_id},
            },
        ])
        return steps

    @staticmethod
    def characterization_study(
        cohort_name: str,
        condition_concepts: list[int],
        source_id: int = 1,
    ) -> list[dict[str, Any]]:
        """Baseline characterization study design."""
        return [
            {
                "tool_name": "create_concept_set",
                "parameters": {
                    "name": f"{cohort_name} Conditions",
                    "concepts": [{"concept_id": c, "include_descendants": True} for c in condition_concepts],
                },
            },
            {
                "tool_name": "create_cohort_definition",
                "parameters": {
                    "name": cohort_name,
                    "expression_json": {
                        "PrimaryCriteria": {
                            "CriteriaList": [{"ConditionOccurrence": {"CodesetId": 0, "First": True}}],
                            "ObservationWindow": {"PriorDays": 365, "PostDays": 0},
                        },
                        "QualifiedLimit": {"Type": "First"},
                    },
                },
            },
            {
                "tool_name": "generate_cohort",
                "parameters": {"source_id": source_id},
            },
            {
                "tool_name": "run_characterization",
                "parameters": {"source_id": source_id},
            },
        ]

    @staticmethod
    def list_templates() -> list[dict[str, str]]:
        return [
            {"name": "incident_cohort", "description": "New user / incident cohort with optional drug exposure"},
            {"name": "characterization_study", "description": "Baseline characterization of a condition cohort"},
        ]

    @staticmethod
    def format_for_prompt() -> str:
        lines = ["Available workflow templates:"]
        for t in WorkflowTemplates.list_templates():
            lines.append(f"  - {t['name']}: {t['description']}")
        return "\n".join(lines)
```
